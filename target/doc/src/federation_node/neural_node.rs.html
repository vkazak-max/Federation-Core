22pub const HIDDEN_SIZE: usize = 8;
23pub const OUTPUT_SIZE: usize = 5;
24pub const LEARNING_RATE: f64 = 0.15;
25pub const MOMENTUM: f64 = 0.5;

33fn relu(x: f64) -> f64 { x.max(0.0) }
34fn relu_derivative(x: f64) -> f64 { if x > 0.0 { 1.0 } else { 0.0 } }
35fn sigmoid(x: f64) -> f64 { 1.0 / (1.0 + (-x).exp()) }
36fn softmax(v: &[f64]) -> Vec<f64> {
37    let max = v.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
38    let exps: Vec<f64> = v.iter().map(|x| (x - max).exp()).collect();
39    let sum: f64 = exps.iter().sum();
40    exps.iter().map(|e| e / sum).collect()
41}

47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct NeuralWeights {
49    pub weights: Vec<Vec<f64>>,
50    pub biases: Vec<f64>,
51    pub velocity: Vec<Vec<f64>>,
52    pub bias_velocity: Vec<f64>,
53}
54
55impl NeuralWeights {
56    pub fn new(input: usize, output: usize, seed: u64) -> Self {
57        let mut rng = seed;
58        let weights = (0..output).map(|_| {
59            (0..input).map(|_| {
60                rng ^= rng << 13; rng ^= rng >> 7; rng ^= rng << 17;
61                ((rng as i64 % 1000) as f64) / 5000.0
62            }).collect()
63        }).collect();
64        let biases = vec![0.0; output];
65        let velocity = vec![vec![0.0; input]; output];
66        let bias_velocity = vec![0.0; output];
67        NeuralWeights { weights, biases, velocity, bias_velocity }
68    }
69
70    pub fn forward(&self, input: &[f64]) -> Vec<f64> {
71        self.weights.iter().zip(self.biases.iter()).map(|(row, bias)| {
72            row.iter().zip(input.iter()).map(|(w, x)| w * x).sum::<f64>() + bias
73        }).collect()
74    }
75
76    pub fn update(&mut self, grad_w: &[Vec<f64>], grad_b: &[f64]) {
77        for i in 0..self.weights.len() {
78            for j in 0..self.weights[i].len() {
79                self.velocity[i][j] = MOMENTUM * self.velocity[i][j]
80                    - LEARNING_RATE * grad_w[i][j];
81                self.weights[i][j] += self.velocity[i][j];
82            }
83            self.bias_velocity[i] = MOMENTUM * self.bias_velocity[i]
84                - LEARNING_RATE * grad_b[i];
85            self.biases[i] += self.bias_velocity[i];
86        }
87    }
88}

95pub struct NeuralState {
96    pub node_id: String,
97    pub layer1: NeuralWeights,
98    pub layer2: NeuralWeights,
99    pub training_steps: u64,
100    pub total_loss: f64,
101    pub success_rate: f64,
102    pub neighbor_weights: HashMap<String, f64>,
103    pub congestion_history: Vec<f64>,
104    pub last_prediction: Option<CongestionPrediction>,
105}

107impl NeuralState {
108    pub fn new(node_id: &str) -> Self {
109        let mut h: u64 = 0xcbf29ce484222325;
110        for b in node_id.bytes() { h ^= b as u64; h = h.wrapping_mul(0x100000001b3); }
111        NeuralState {
112            node_id: node_id.to_string(),
113            layer1: NeuralWeights::new(INPUT_SIZE, HIDDEN_SIZE, h),
114            layer2: NeuralWeights::new(HIDDEN_SIZE, OUTPUT_SIZE, h ^ 0xdeadbeef),
115            training_steps: 0,
116            total_loss: 0.0,
117            success_rate: 0.5,
118            neighbor_weights: HashMap::new(),
119            congestion_history: vec![],
120            last_prediction: None,
121        }
122    }
123
124    /// –ü—Ä—è–º–æ–π –ø—Ä–æ—Ö–æ–¥: –≤—Ö–æ–¥–Ω–æ–π –≤–µ–∫—Ç–æ—Ä ‚Üí [route_weight, congestion_prob, quality_score]
125    pub fn forward(&self, input: &NeuralInput) -> NeuralOutput {
126        let x = input.to_vector();
127        let h1: Vec<f64> = self.layer1.forward(&x).iter().map(|&v| relu(v)).collect();
128        let out = self.layer2.forward(&h1);
129        let probs = softmax(&out);
130        let congestion  = sigmoid(out[1]);
131        let decoy       = sigmoid(out[3]);
132        let strike      = sigmoid(out[4]);
133        NeuralOutput {
134            route_weight:    sigmoid(out[0]),
135            congestion_prob: congestion,
136            quality_score:   sigmoid(out[2]),
137            decoy_intensity: decoy,
138            strike_focus:    strike,
139            softmax_probs:   probs,
140            hidden_state:    h1,
141            tactic:          NeuralTactic::decide(congestion, decoy, strike),
142        }
143    }
144
145    /// –û–±—É—á–µ–Ω–∏–µ –Ω–∞ —É—Å–ø–µ—Ö–µ: –ø–∞–∫–µ—Ç –¥–æ—à—ë–ª ‚Üí –∑–∞–∫—Ä–µ–ø–ª—è–µ–º –ø—É—Ç—å
146    pub fn backpropagate_success(&mut self, input: &NeuralInput,
147        target: &NeuralTarget, neighbor_id: &str) {
148        let x = input.to_vector();
149        let h1_raw = self.layer1.forward(&x);
150        let h1: Vec<f64> = h1_raw.iter().map(|&v| relu(v)).collect();
151        let out = self.layer2.forward(&h1);
152
153        // Loss = MSE –º–µ–∂–¥—É –≤—ã—Ö–æ–¥–æ–º –∏ —Ü–µ–ª–µ–≤—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º
154        let target_vec = target.to_vector();
155        let loss: f64 = out.iter().zip(target_vec.iter())
156            .map(|(o, t)| (o - t).powi(2)).sum::<f64>() / OUTPUT_SIZE as f64;
157        self.total_loss = self.total_loss * 0.99 + loss * 0.01;
158
159        // –ì—Ä–∞–¥–∏–µ–Ω—Ç output —Å–ª–æ—è: Œ¥ = 2*(out - target) / N
160        let delta2: Vec<f64> = out.iter().zip(target_vec.iter())
161            .map(|(o, t)| 2.0 * (o - t) / OUTPUT_SIZE as f64).collect();
162
163        // –ì—Ä–∞–¥–∏–µ–Ω—Ç –≤–µ—Å–æ–≤ layer2: dL/dW2 = Œ¥2 ‚äó h1
164        let grad_w2: Vec<Vec<f64>> = delta2.iter()
165            .map(|d| h1.iter().map(|h| d * h).collect()).collect();
166        let grad_b2: Vec<f64> = delta2.clone();
167
168        // Backprop —á–µ—Ä–µ–∑ hidden: Œ¥1 = (W2^T ¬∑ Œ¥2) * relu'(h1_raw)
169        let mut delta1 = vec![0.0; HIDDEN_SIZE];
170        for j in 0..HIDDEN_SIZE {
171            for k in 0..OUTPUT_SIZE {
172                delta1[j] += self.layer2.weights[k][j] * delta2[k];
173            }
174            delta1[j] *= relu_derivative(h1_raw[j]);
175        }
176
177        // –ì—Ä–∞–¥–∏–µ–Ω—Ç –≤–µ—Å–æ–≤ layer1: dL/dW1 = Œ¥1 ‚äó x
178        let grad_w1: Vec<Vec<f64>> = delta1.iter()
179            .map(|d| x.iter().map(|xi| d * xi).collect()).collect();
180        let grad_b1: Vec<f64> = delta1;
181
182        // –û–±–Ω–æ–≤–ª—è–µ–º –≤–µ—Å–∞
183        self.layer1.update(&grad_w1, &grad_b1);
184        self.layer2.update(&grad_w2, &grad_b2);
185
186        // –û–±–Ω–æ–≤–ª—è–µ–º –≤–µ—Å —Å–æ—Å–µ–¥–∞
187        let reward = if target.success { 0.1 } else { -0.05 };
188        let w = self.neighbor_weights.entry(neighbor_id.to_string()).or_insert(0.5);
189        *w = (*w + reward).clamp(0.0, 1.0);
190
191        // –û–±–Ω–æ–≤–ª—è–µ–º success rate
192        if target.success {
193            self.success_rate = self.success_rate * 0.95 + 0.05;
194        } else {
195            self.success_rate *= 0.95;
196        }
197
198        self.training_steps += 1;
199    }
200
201    /// –ü—Ä–µ–¥—Å–∫–∞–∑–∞—Ç—å –∑–∞—Ç–æ—Ä –Ω–∞ –æ—Å–Ω–æ–≤–µ –∏—Å—Ç–æ—Ä–∏–∏ –∑–∞–¥–µ—Ä–∂–µ–∫
202    pub fn predict_congestion(&mut self, current_latency_ms: f64) -> CongestionPrediction {
203        self.congestion_history.push(current_latency_ms);
204        if self.congestion_history.len() > CONGESTION_WINDOW {
205            self.congestion_history.remove(0);
206        }
207
208        let history = &self.congestion_history;
209        if history.len() < 3 {
210            return CongestionPrediction {
211                probability: 0.0, trend: Trend::Stable,
212                predicted_latency_ms: current_latency_ms,
213                confidence: 0.1, action: "–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –¥–∞–Ω–Ω—ã—Ö".into(),
214            };
215        }
216
217        // –õ–∏–Ω–µ–π–Ω—ã–π —Ç—Ä–µ–Ω–¥: –Ω–∞–∫–ª–æ–Ω —Å–∫–æ–ª—å–∑—è—â–µ–≥–æ —Å—Ä–µ–¥–Ω–µ–≥–æ
218        let n = history.len() as f64;
219        let mean = history.iter().sum::<f64>() / n;
220        let slope = history.iter().enumerate()
221            .map(|(i, &v)| (i as f64 - n / 2.0) * (v - mean))
222            .sum::<f64>() / history.iter().enumerate()
223            .map(|(i, _)| (i as f64 - n / 2.0).powi(2))
224            .sum::<f64>().max(1e-10);
225
226        // –ù–µ–π—Ä–æ–Ω–Ω–∞—è –æ—Ü–µ–Ω–∫–∞ —á–µ—Ä–µ–∑ forward pass
227        let norm_latency = (current_latency_ms / 200.0).min(1.0);
228        let _norm_slope = (slope / 50.0).clamp(-1.0, 1.0);
229        let std_dev = {
230            let var = history.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / n;
231            var.sqrt() / 100.0
232        };
233
234        let neural_input = NeuralInput {
235            latency: norm_latency,
236            bandwidth: 1.0 - norm_latency,
237            reliability: 1.0 - std_dev.min(1.0),
238            trust: self.success_rate,
239            ethics_score: 1.0,
240        };
241        let neural_out = self.forward(&neural_input);
242
243        let base_prob = neural_out.congestion_prob;
244        let trend_factor = if slope > 5.0 { slope / 50.0 } else { 0.0 };
245        let probability = (base_prob + trend_factor * 0.3).clamp(0.0, 1.0);
246
247        let trend = if slope > 5.0 { Trend::Rising }
248            else if slope < -5.0 { Trend::Falling }
249            else { Trend::Stable };
250
251        let predicted_latency = current_latency_ms + slope * 5.0;
252        let confidence = (history.len() as f64 / CONGESTION_WINDOW as f64).min(1.0);
253
254        let action = if probability > CONGESTION_THRESHOLD {
255            format!("‚ö†Ô∏è  –ó–ê–¢–û–† –ü–†–ï–î–°–ö–ê–ó–ê–ù! –ü–µ—Ä–µ–∫–ª—é—á–∏—Ç—å –º–∞—Ä—à—Ä—É—Ç (prob={:.1}%)", probability * 100.0)
256        } else if probability > 0.4 {
257            format!("üìà –ù–∞–≥—Ä—É–∑–∫–∞ —Ä–∞—Å—Ç—ë—Ç, –º–æ–Ω–∏—Ç–æ—Ä–∏–Ω–≥ (prob={:.1}%)", probability * 100.0)
258        } else {
259            format!("‚úÖ –°–µ—Ç—å —Å—Ç–∞–±–∏–ª—å–Ω–∞ (prob={:.1}%)", probability * 100.0)
260        };
261
262        let pred = CongestionPrediction {
263            probability, trend, predicted_latency_ms: predicted_latency,
264            confidence, action,
265        };
266        self.last_prediction = Some(pred.clone());
267        pred
268    }
269}

311impl NeuralTactic {
312    /// –¢–∞–∫—Ç–∏–∫–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –ø–æ –≤—Ö–æ–¥–Ω–æ–º—É –≤–µ–∫—Ç–æ—Ä—É –Ω–∞–ø—Ä—è–º—É—é
313    /// neural outputs –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –∫–∞–∫ –≤–µ—Å–∞ —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç–∏
314    pub fn decide(congestion: f64, decoy: f64, strike: f64) -> Self {
315        // –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–Ω—Ç—Ä–∞ 0.5
316        let d = decoy - 0.5;    // –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ decoy –æ—Ç –Ω–µ–π—Ç—Ä–∞–ª–∏
317        let s = strike - 0.5;   // –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ strike –æ—Ç –Ω–µ–π—Ç—Ä–∞–ª–∏
318        let c = congestion - 0.5;
319
320        // –í—ã–±–∏—Ä–∞–µ–º –ø–æ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–º—É –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—é
321        if c > 0.1 && s > 0.05 {
322            NeuralTactic::AikiReflection
323        } else if s > 0.1 && d > 0.1 {
324            NeuralTactic::Hybrid
325        } else if s > d && s > 0.05 {
326            NeuralTactic::CumulativeStrike
327        } else if d > s && d > 0.05 {
328            NeuralTactic::StandoffDecoy
329        } else {
330            NeuralTactic::Passive
331        }
332    }
333
334    /// –¢–∞–∫—Ç–∏–∫–∞ —Å —É—á—ë—Ç–æ–º –≤—Ö–æ–¥–Ω–æ–≥–æ –≤–µ–∫—Ç–æ—Ä–∞ (–±–æ–ª–µ–µ —Ç–æ—á–Ω–∞—è)
335    pub fn decide_from_input(latency: f64, congestion: f64,
336        decoy: f64, strike: f64) -> Self {
337        // –õ–∞—Ç–µ–Ω—Ç–Ω–æ—Å—Ç—å –Ω–∞–ø—Ä—è–º—É—é –≥–æ–≤–æ—Ä–∏—Ç –æ–± —É–≥—Ä–æ–∑–µ
338        if latency > 0.80 && congestion > 0.60 {
339            NeuralTactic::AikiReflection
340        } else if latency > 0.70 && decoy > 0.50 {
341            NeuralTactic::StandoffDecoy
342        } else if latency < 0.50 && strike > 0.50 {
343            NeuralTactic::CumulativeStrike
344        } else if latency > 0.40 && decoy > 0.50 && strike > 0.50 {
345            NeuralTactic::Hybrid
346        } else {
347            NeuralTactic::Passive
348        }
349    }
350    pub fn name(&self) -> &str {
351        match self {
352            NeuralTactic::Passive          => "Passive",
353            NeuralTactic::StandoffDecoy    => "StandoffDecoy",
354            NeuralTactic::CumulativeStrike => "CumulativeStrike",
355            NeuralTactic::AikiReflection   => "AikiReflection",
356            NeuralTactic::Hybrid           => "Hybrid",
357        }
358    }
359}

440    /// –û—Ü–µ–Ω–∏—Ç—å –º–∞—Ä—à—Ä—É—Ç —á–µ—Ä–µ–∑ –Ω–µ–π—Ä–æ–Ω–Ω—É—é —Å–µ—Ç—å
441    pub fn score_route(&mut self, neighbor_id: &str, input: &NeuralInput) -> NeuralOutput {
442        let state = self.states.entry(neighbor_id.to_string())
443            .or_insert_with(|| NeuralState::new(neighbor_id));
444        self.routes_computed += 1;
445        state.forward(input)
446    }
447
448    /// –í—ã–±—Ä–∞—Ç—å –ª—É—á—à–∏–π –º–∞—Ä—à—Ä—É—Ç –∏–∑ –∫–∞–Ω–¥–∏–¥–∞—Ç–æ–≤
449    pub fn select_best(&mut self, candidates: Vec<(String, NeuralInput)>) -> Option<String> {
450        if candidates.is_empty() { return None; }
451        let scored: Vec<(String, f64)> = candidates.iter().map(|(id, input)| {
452            let state = self.states.entry(id.clone())
453                .or_insert_with(|| NeuralState::new(id));
454            let out = state.forward(input);
455            let score = out.route_weight * 0.5
456                + out.quality_score * 0.3
457                + (1.0 - out.congestion_prob) * 0.2;
458            let neighbor_bonus = *state.neighbor_weights.get(id).unwrap_or(&0.5);
459            (id.clone(), score + neighbor_bonus * 0.1)
460        }).collect();
461
462        scored.into_iter().max_by(|a, b| a.1.partial_cmp(&b.1).unwrap())
463            .map(|(id, _)| id)
464    }
465
466    /// –û–±—É—á–∏—Ç—å —Å–µ—Ç—å –Ω–∞ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–µ –¥–æ—Å—Ç–∞–≤–∫–∏
467    pub fn train_on_delivery(&mut self, neighbor_id: &str,
468        input: &NeuralInput, success: bool, quality: f64) {
469        let target = if success {
470            NeuralTarget::success_route(quality)
471        } else {
472            NeuralTarget::failed_route()
473        };
474        let state = self.states.entry(neighbor_id.to_string())
475            .or_insert_with(|| NeuralState::new(neighbor_id));
476        state.backpropagate_success(input, &target, neighbor_id);
477        if success { self.routes_improved += 1; }
478    }
479
480    pub fn stats(&self) -> RouterNeuralStats {
481        let avg_success = if self.states.is_empty() { 0.0 } else {
482            self.states.values().map(|s| s.success_rate).sum::<f64>()
483                / self.states.len() as f64
484        };
485        RouterNeuralStats {
486            nodes_tracked: self.states.len(),
487            routes_computed: self.routes_computed,
488            routes_improved: self.routes_improved,
489            avg_success_rate: avg_success,
490            training_steps: self.states.values().map(|s| s.training_steps).sum(),
491        }
492    }
493}

543    pub cpu_cores: u8,
544    pub cpu_load: f64,        // 0.0-1.0
545    pub ram_total_mb: u32,
546    pub ram_used_mb: u32,

553impl ResourceProfile {
554    pub fn ram_load(&self) -> f64 {
555        self.ram_used_mb as f64 / self.ram_total_mb.max(1) as f64
556    }
557    pub fn is_throttling(&self) -> bool {
558        self.temp_celsius >= THERMAL_THROTTLE
559    }
560    pub fn is_low_battery(&self) -> bool {
561        self.battery_pct.map(|b| b < BATTERY_LOW).unwrap_or(false)
562    }
563    pub fn compute_score(&self) -> f64 {
564        // –ù–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω–∞—è –≤—ã—á–∏—Å–ª–∏—Ç–µ–ª—å–Ω–∞—è –º–æ—â–Ω–æ—Å—Ç—å 0.0-1.0
565        let cpu_avail  = (1.0 - self.cpu_load).max(0.0);
566        let ram_avail  = (1.0 - self.ram_load()).max(0.0);
567        let thermal_ok = if self.is_throttling() { 0.5 } else { 1.0 };
568        let battery_ok = if self.is_low_battery() { 0.3 } else { 1.0 };
569        cpu_avail * 0.4 + ram_avail * 0.3 + thermal_ok * 0.2 + battery_ok * 0.1
570    }

666pub struct AdaptiveScheduler {

668    pub profile: ResourceProfile,
669    pub budget: ComputeBudget,
670    pub scheduled: Vec<AdaptiveTask>,
671    pub skipped: Vec<AdaptiveTask>,
672}
673
674impl AdaptiveScheduler {
675    pub fn new(profile: ResourceProfile) -> Self {
676        let budget = ComputeBudget::from_profile(&profile);
677        AdaptiveScheduler {
678            node_id: profile.node_id.clone(),
679            profile, budget,
680            scheduled: vec![], skipped: vec![],
681        }
682    }
683
684    pub fn schedule(&mut self, tasks: Vec<AdaptiveTask>) {
685        self.scheduled.clear();
686        self.skipped.clear();
687
688        let budget_level = match self.budget {
689            ComputeBudget::Full      => 4u8,
690            ComputeBudget::Reduced   => 3,
691            ComputeBudget::Minimal   => 2,
692            ComputeBudget::Emergency => 1,
693        };
694
695        let mut sorted = tasks;
696        sorted.sort_by(|a,b| b.priority.cmp(&a.priority));
697
698        let mut total_cpu = 0.0f64;
699        for task in sorted {
700            let task_level = match task.required_budget {
701                ComputeBudget::Emergency => 1u8,
702                ComputeBudget::Minimal   => 2,
703                ComputeBudget::Reduced   => 3,
704                ComputeBudget::Full      => 4,
705            };
706            let cpu_ok = total_cpu + task.cpu_weight <= 0.90;
707            if task_level <= budget_level && cpu_ok {
708                total_cpu += task.cpu_weight;
709                self.scheduled.push(task);
710            } else {
711                self.skipped.push(task);
712            }
713        }
714    }
715
716    pub fn stats(&self) -> SchedulerStats {
717        SchedulerStats {
718            node_id: self.node_id.clone(),
719            budget: self.budget.clone(),
720            compute_score: self.profile.compute_score(),
721            scheduled_count: self.scheduled.len(),
722            skipped_count: self.skipped.len(),
723            cpu_load: self.profile.cpu_load,
724            ram_load: self.profile.ram_load(),
725            is_throttling: self.profile.is_throttling(),
726            inference_interval_ms: self.budget.inference_interval_ms(),
727        }
728    }
729}
