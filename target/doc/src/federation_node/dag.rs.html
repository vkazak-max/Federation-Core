38/// Одна запись в DAG — факт маршрутизации пакета через узел
39#[derive(Debug, Clone, Serialize, Deserialize)]
40pub struct DagNode {
41    /// Уникальный ID вершины (hash от содержимого)
42    pub id: String,
43    /// Узел который создал эту запись
44    pub reporter_id: String,
45    /// Маршрут который был использован
46    pub route_path: Vec<String>,
47    /// Снимок SSAU тензоров на момент маршрутизации
48    pub ssau_snapshot: Vec<SsauSnapshot>,
49    /// Суммарная задержка маршрута (мс)
50    pub total_latency_ms: f64,
51    /// Время создания (Unix ms)
52    pub timestamp: i64,
53    /// ID родительских вершин (предыдущие записи)
54    pub parents: Vec<String>,
55    /// Глубина в графе
56    pub depth: usize,
57    /// Статус верификации
58    pub verified: bool,
59    /// Начисленная награда PoA
60    pub poa_reward: f64,
61    /// Оценка честности (0.0 = ложь, 1.0 = правда)
62    pub honesty_score: f64,
63}

65/// Снимок одного тензора для записи в DAG
66#[derive(Debug, Clone, Serialize, Deserialize)]
67pub struct SsauSnapshot {
68    pub from_node: String,
69    pub to_node: String,
70    pub latency_ms: f64,
71    pub bandwidth_mbps: f64,
72    pub reliability: f64,
73}
74
75impl From<&SsauTensor> for SsauSnapshot {
76    fn from(t: &SsauTensor) -> Self {
77        SsauSnapshot {
78            from_node: t.from_node.clone(),
79            to_node: t.to_node.clone(),
80            latency_ms: t.latency.mean,
81            bandwidth_mbps: t.bandwidth,
82            reliability: t.reliability,
83        }
84    }
85}

87impl DagNode {
88    /// Создать новую вершину DAG
89    pub fn new(
90        reporter_id: &str,
91        route_path: Vec<String>,
92        tensors: &[&SsauTensor],
93        parents: Vec<String>,
94        depth: usize,
95    ) -> Self {
96        use std::time::{SystemTime, UNIX_EPOCH};
97        let now = SystemTime::now()
98            .duration_since(UNIX_EPOCH)
99            .unwrap()
100            .as_millis() as i64;
101
102        let total_latency: f64 = tensors.iter().map(|t| t.latency.mean).sum();
103        let snapshots: Vec<SsauSnapshot> = tensors.iter().map(|t| SsauSnapshot::from(*t)).collect();
104
105        // Простой hash: комбинируем reporter + timestamp + path
106        let id = format!("{:x}", {
107            let mut h: u64 = 0xcbf29ce484222325;
108            for b in format!("{}{}{}",
109                reporter_id,
110                now,
111                route_path.join(",")
112            ).bytes() {
113                h ^= b as u64;
114                h = h.wrapping_mul(0x100000001b3);
115            }
116            h
117        });
118
119        DagNode {
120            id,
121            reporter_id: reporter_id.to_string(),
122            route_path,
123            ssau_snapshot: snapshots,
124            total_latency_ms: total_latency,
125            timestamp: now,
126            parents,
127            depth,
128            verified: false,
129            poa_reward: 0.0,
130            honesty_score: 1.0,
131        }
132    }
133}

153/// Результат расчёта награды PoA
154#[derive(Debug, Clone, Serialize, Deserialize)]
155pub struct PoaResult {
156    pub node_id: String,
157    pub reward: f64,
158    pub penalty: f64,
159    pub net: f64,
160    pub reason: String,
161    pub honesty_score: f64,
162}
163
164/// Рассчитать награду Proof-of-Awareness для узла.
165///
166/// Логика:
167///   - Базовая награда за участие в маршрутизации
168///   - Бонус за высокое качество данных SSAU
169///   - Штраф если Triangle Check выявил ложь
170///   - Множитель от текущего Trust Weight
171pub fn calculate_poa_reward(
172    node: &DagNode,
173    trust_registry: &TrustRegistry,
174    witness_latencies: Option<(f64, f64)>, // (L_AC, L_BC) от соседей-свидетелей
175) -> PoaResult {
176    let trust_weight = trust_registry.get_trust(&node.reporter_id);
177    let mut reward = BASE_POA_REWARD * trust_weight;
178    let mut penalty = 0.0;
179    let mut honesty_score = 1.0;
180    let mut reason = String::new();
181
182    // Бонус за качество канала
183    let avg_reliability: f64 = if node.ssau_snapshot.is_empty() {
184        0.0
185    } else {
186        node.ssau_snapshot.iter().map(|s| s.reliability).sum::<f64>()
187            / node.ssau_snapshot.len() as f64
188    };
189
190    if avg_reliability > 0.95 {
191        reward += QUALITY_BONUS;
192        reason += &format!("Quality bonus +{:.2} (reliability={:.3}). ", QUALITY_BONUS, avg_reliability);
193    }
194
195    // Triangle Check если есть свидетели
196    if let Some((l_ac, l_bc)) = witness_latencies {
197        if let Some(snapshot) = node.ssau_snapshot.first() {
198            let check = triangle_check(
199                snapshot.latency_ms,
200                l_ac,
201                l_bc,
202                trust_weight,
203            );
204
205            if !check.passed {
206                penalty = DISHONESTY_PENALTY * check.deviation_score;
207                honesty_score = 1.0 - check.deviation_score;
208                reason += &format!(
209                    "⛔ Triangle Check FAILED! Penalty -{:.2}. Deviation={:.1}%. ",
210                    penalty,
211                    check.deviation_score * 100.0
212                );
213            } else {
214                reason += "✅ Triangle Check passed. ";
215                honesty_score = 1.0;
216            }
217        }
218    }
219
220    // Бонус за скорость (задержка < 20ms)
221    if node.total_latency_ms < 20.0 {
222        let speed_bonus = 0.3 * (1.0 - node.total_latency_ms / 20.0);
223        reward += speed_bonus;
224        reason += &format!("Speed bonus +{:.2} (latency={:.1}ms). ", speed_bonus, node.total_latency_ms);
225    }
226
227    let net = (reward - penalty).max(0.0);
228    reason += &format!("Net: {:.4} монет.", net);
229
230    PoaResult {
231        node_id: node.reporter_id.clone(),
232        reward,
233        penalty,
234        net,
235        reason,
236        honesty_score,
237    }
238}

244/// Асинхронный DAG Федерации.
245/// В Phase 2 (light version) — in-memory.
246/// В Phase 3 — персистентное хранилище.
247#[derive(Debug, Default, Serialize, Deserialize)]
248pub struct FederationDag {
249    /// Все вершины: id → DagNode
250    pub nodes: HashMap<String, DagNode>,
251    /// Все рёбра
252    pub edges: Vec<DagEdge>,
253    /// Баланс монет: node_id → накопленные монеты
254    pub balances: HashMap<String, f64>,
255    /// Tips — вершины без исходящих рёбер (кончики DAG)
256    pub tips: Vec<String>,
257    /// Счётчик операций
258    pub total_operations: u64,
259}

261impl FederationDag {
262    pub fn new() -> Self {
263        Self::default()
264    }

266    /// Добавить новую запись маршрута в DAG.
267    ///
268    /// Алгоритм:
269    ///   1. Выбираем 2 tips как родителей (подтверждаем их)
270    ///   2. Создаём новую вершину
271    ///   3. Рассчитываем PoA награду
272    ///   4. Обновляем tips
273    ///   5. Начисляем монеты
274    pub fn append_route(
275        &mut self,
276        reporter_id: &str,
277        route_path: Vec<String>,
278        tensors: &[&SsauTensor],
279        trust_registry: &mut TrustRegistry,
280        witness_latencies: Option<(f64, f64)>,
281    ) -> (DagNode, PoaResult) {
282        // Выбираем родителей из текущих tips (подтверждаем их)
283        let parents: Vec<String> = self.tips.iter().take(2).cloned().collect();
284        let depth = if parents.is_empty() {
285            0
286        } else {
287            parents.iter()
288                .filter_map(|p| self.nodes.get(p))
289                .map(|n| n.depth)
290                .max()
291                .unwrap_or(0) + 1
292        };
293
294        // Создаём вершину
295        let mut node = DagNode::new(reporter_id, route_path, tensors, parents.clone(), depth);
296
297        // Рассчитываем PoA награду
298        let poa = calculate_poa_reward(&node, trust_registry, witness_latencies);
299        node.poa_reward = poa.net;
300        node.honesty_score = poa.honesty_score;
301        node.verified = poa.honesty_score > 0.5;
302
303        // Добавляем рёбра от родителей к новой вершине
304        use std::time::{SystemTime, UNIX_EPOCH};
305        let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_millis() as i64;
306        for parent_id in &parents {
307            self.edges.push(DagEdge {
308                from_id: parent_id.clone(),
309                to_id: node.id.clone(),
310                confirmed_by: reporter_id.to_string(),
311                timestamp: now,
312            });
313            // Убираем подтверждённый tip
314            self.tips.retain(|t| t != parent_id);
315        }
316
317        // Начисляем монеты
318        *self.balances.entry(reporter_id.to_string()).or_insert(0.0) += poa.net;
319
320        // Добавляем новую вершину как tip
321        let node_id = node.id.clone();
322        self.nodes.insert(node_id.clone(), node.clone());
323        self.tips.push(node_id);
324        self.total_operations += 1;
325
326        // Ограничиваем размер DAG
327        if self.nodes.len() > MAX_DAG_DEPTH {
328            self.prune_old_nodes();
329        }
330
331        (node, poa)
332    }

334    /// Получить историю маршрутов конкретного узла
335    pub fn get_node_history(&self, node_id: &str) -> Vec<&DagNode> {
336        let mut history: Vec<&DagNode> = self.nodes.values()
337            .filter(|n| n.reporter_id == node_id)
338            .collect();
339        history.sort_by_key(|n| n.timestamp);
340        history
341    }

343    /// Верифицировать цепочку от вершины до genesis
344    pub fn verify_chain(&self, node_id: &str) -> bool {
345        let mut current_id = node_id.to_string();
346        let mut visited = std::collections::HashSet::new();
347
348        loop {
349            if visited.contains(&current_id) {
350                return false; // Цикл — невалидно (DAG не должен иметь циклов)
351            }
352            visited.insert(current_id.clone());
353
354            match self.nodes.get(&current_id) {
355                None => return false,
356                Some(node) => {
357                    if node.parents.is_empty() {
358                        return true; // Дошли до genesis
359                    }
360                    if !node.verified {
361                        return false;
362                    }
363                    current_id = node.parents[0].clone();
364                }
365            }
366        }
367    }
368
369    /// Удалить старые подтверждённые вершины (оставить только последние)
370    fn prune_old_nodes(&mut self) {
371        let mut nodes_by_depth: Vec<(String, usize)> = self.nodes.iter()
372            .map(|(id, n)| (id.clone(), n.depth))
373            .collect();
374        nodes_by_depth.sort_by_key(|(_, d)| *d);
375
376        // Удаляем самые старые (низкая глубина), оставляем MAX_DAG_DEPTH/2
377        let to_remove = nodes_by_depth.len() - MAX_DAG_DEPTH / 2;
378        for (id, _) in nodes_by_depth.iter().take(to_remove) {
379            if !self.tips.contains(id) {
380                self.nodes.remove(id);
381            }
382        }
383    }

385    /// Статистика DAG
386    pub fn stats(&self) -> DagStats {
387        let total_rewards: f64 = self.balances.values().sum();
388        let verified_count = self.nodes.values().filter(|n| n.verified).count();
389        let avg_honesty = if self.nodes.is_empty() { 1.0 } else {
390            self.nodes.values().map(|n| n.honesty_score).sum::<f64>() / self.nodes.len() as f64
391        };
392
393        DagStats {
394            total_nodes: self.nodes.len(),
395            total_edges: self.edges.len(),
396            tips_count: self.tips.len(),
397            total_operations: self.total_operations,
398            total_rewards_issued: total_rewards,
399            verified_nodes: verified_count,
400            avg_honesty_score: avg_honesty,
401            richest_node: self.balances.iter()
402                .max_by(|a, b| a.1.partial_cmp(b.1).unwrap())
403                .map(|(id, bal)| format!("{}: {:.4} монет", id, bal))
404                .unwrap_or_else(|| "нет данных".to_string()),
405        }
406    }
407}
