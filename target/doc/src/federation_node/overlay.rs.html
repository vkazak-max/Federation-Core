93    /// –ü–æ–¥–∫–ª—é—á–∏—Ç—å—Å—è –∫ seed-—É–∑–ª–∞–º
94    pub async fn bootstrap(&mut self, node: Arc<FederationNode>) -> usize {
95        let mut connected = 0;
96
97        for seed in &self.seeds {
98            log::info!("üå± Bootstrap: –ø–æ–¥–∫–ª—é—á–∞–µ–º—Å—è –∫ seed {} ({})", seed.node_id, seed.address);
99
100            match tokio::time::timeout(
101                Duration::from_secs(5),
102                node.clone().connect_to_peer(&seed.address),
103            ).await {
104                Ok(Ok(peer_id)) => {
105                    log::info!("‚úÖ Bootstrap: –ø–æ–¥–∫–ª—é—á–∏–ª–∏—Å—å –∫ {}", peer_id);
106                    self.connected_seeds.push(peer_id);
107                    connected += 1;
108                }
109                Ok(Err(e)) => {
110                    log::warn!("‚ö†Ô∏è Bootstrap: –Ω–µ —É–¥–∞–ª–æ—Å—å –ø–æ–¥–∫–ª—é—á–∏—Ç—å—Å—è –∫ {}: {}", seed.node_id, e);
111                }

114                }
115            }
116        }
117
118        self.bootstrap_complete = connected > 0;
119        connected
120    }

293    pub async fn send_onion(
294        self: Arc<Self>,
295        route: Vec<String>,
296        payload: &[u8],
297    ) -> Result<String, String> {
298        if route.len() < 2 {
299            return Err("–ú–∞—Ä—à—Ä—É—Ç –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å –º–∏–Ω–∏–º—É–º 2 —É–∑–ª–∞".to_string());
300        }
301
302        let (packet, _keys) = OnionBuilder::new()
303            .with_route(route.clone())
304            .build(payload)
305            .map_err(|e| e.to_string())?;
306
307        // –ü—Ä–æ–≤–µ—Ä—è–µ–º nullifier (anti-replay)
308        let nullifier = packet.outer_layer.nullifier.clone();
309        let mut nulls = self.nullifiers.lock().await;
310        if !nulls.check_and_add(&nullifier) {
311            return Err("Replay attack detected ‚Äî –ø–∞–∫–µ—Ç –æ—Ç–±—Ä–æ—à–µ–Ω".to_string());
312        }
313        drop(nulls);
314
315        *self.packets_onion_wrapped.lock().await += 1;
316
317        Ok(format!(
318            "Onion –ø–∞–∫–µ—Ç –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω: {} —Å–ª–æ—ë–≤, –º–∞—Ä—à—Ä—É—Ç {:?}, nullifier: {}",
319            packet.layer_count,
320            route,
321            &nullifier[..8]
322        ))
323    }

325    /// –í—ã—á–∏—Å–ª–∏—Ç—å –æ–ø—Ç–∏–º–∞–ª—å–Ω—ã–π –º–∞—Ä—à—Ä—É—Ç —á–µ—Ä–µ–∑ AI Router
326    pub async fn compute_route(
327        self: Arc<Self>,
328        destination: &str,
329        priorities: UserPriorities,
330    ) -> Option<Vec<String>> {
331        let ssau_table = self.node.ssau_table.read().await;
332        let trust = self.trust.read().await;
333        let our_id = &self.node.config.node_id.clone();
334
335        let candidates = build_route_candidates(
336            &ssau_table, our_id, destination, &trust, 5
337        );
338
339        if candidates.is_empty() {
340            return None;
341        }
342
343        let mut router = self.router.lock().await;
344        let decision = router.select_route(destination, candidates, &priorities);
345        *self.routes_computed.lock().await += 1;
346
347        decision.chosen_route.map(|r| r.path)
348    }
