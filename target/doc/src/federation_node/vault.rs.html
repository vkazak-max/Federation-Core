56#[derive(Debug, Clone, Serialize, Deserialize)]
57pub struct ZkProof {
58    pub proof_hash: String,
59    pub commitment: String,
60    pub nullifier: String,
61    pub valid: bool,
62    pub expires_at: i64,
63}
64
65impl ZkProof {
66    pub fn generate(owner_id: &str, secret: &[u8], rng: &mut u64) -> Self {
67        // –£–ø—Ä–æ—â—ë–Ω–Ω—ã–π ZK ‚Äî –≤ production –∑–∞–º–µ–Ω–∏—Ç—å –Ω–∞ –Ω–∞—Å—Ç–æ—è—â–∏–π ZKP
68        *rng ^= *rng << 13; *rng ^= *rng >> 7; *rng ^= *rng << 17;
69        let commitment = format!("commit_{:016x}", *rng ^ owner_id.len() as u64);
70        *rng ^= *rng << 13; *rng ^= *rng >> 7;
71        let nullifier = format!("null_{:016x}", *rng);
72        let proof_hash = format!("zkp_{:08x}{:08x}",
73            secret.iter().fold(0u32, |a, &b| a.wrapping_add(b as u32)),
74            owner_id.bytes().fold(0u32, |a, b| a.wrapping_mul(31).wrapping_add(b as u32)));
75
76        use std::time::{SystemTime, UNIX_EPOCH};
77        let now = SystemTime::now().duration_since(UNIX_EPOCH)
78            .unwrap().as_millis() as i64;
79
80        ZkProof {
81            proof_hash, commitment, nullifier,
82            valid: true, expires_at: now + 3_600_000, // 1 —á–∞—Å
83        }
84    }
85
86    pub fn verify(&self) -> bool {
87        use std::time::{SystemTime, UNIX_EPOCH};
88        let now = SystemTime::now().duration_since(UNIX_EPOCH)
89            .unwrap().as_millis() as i64;
90        self.valid && now < self.expires_at
91    }
92}

140// GF(256) –∞—Ä–∏—Ñ–º–µ—Ç–∏–∫–∞ –¥–ª—è —Å—Ö–µ–º—ã –®–∞–º–∏—Ä–∞
141fn gf_add(a: u8, b: u8) -> u8 { a ^ b }
142
143fn gf_mul(a: u8, b: u8) -> u8 {
144    let mut p = 0u8;
145    let mut a = a;
146    let mut b = b;
147    for _ in 0..8 {
148        if b & 1 != 0 { p ^= a; }
149        let carry = a & 0x80;
150        a <<= 1;
151        if carry != 0 { a ^= 0x1b; } // –ø—Ä–∏–º–∏—Ç–∏–≤–Ω—ã–π –ø–æ–ª–∏–Ω–æ–º x^8+x^4+x^3+x+1
152        b >>= 1;
153    }
154    p
155}
156
157fn gf_inv(a: u8) -> u8 {
158    if a == 0 { return 0; }
159    let mut result = 1u8;
160    let mut base = a;
161    let mut exp = 254u8; // a^(256-2) = a^(-1) –ø–æ –º–∞–ª–æ–π —Ç–µ–æ—Ä–µ–º–µ –§–µ—Ä–º–∞
162    while exp > 0 {
163        if exp & 1 != 0 { result = gf_mul(result, base); }
164        base = gf_mul(base, base);
165        exp >>= 1;
166    }
167    result
168}

170// –°—Ö–µ–º–∞ –®–∞–º–∏—Ä–∞ (—É–ø—Ä–æ—â—ë–Ω–Ω–∞—è –Ω–∞–¥ GF(256))
171pub struct ShamirScheme;
172
173impl ShamirScheme {
174    /// –†–∞–∑–¥–µ–ª–∏—Ç—å —Å–µ–∫—Ä–µ—Ç –Ω–∞ N –æ—Å–∫–æ–ª–∫–æ–≤, K –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –¥–ª—è –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏—è
175    pub fn split(secret: &[u8], n: usize, k: usize,
176                 rng: &mut u64) -> Vec<Vec<u8>> {
177        let mut shards = vec![vec![0u8; secret.len()]; n];
178
179        // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º k-1 —Å–ª—É—á–∞–π–Ω—ã—Ö –ø–æ–ª–∏–Ω–æ–º–æ–≤
180        let coeffs: Vec<Vec<u8>> = (0..k-1).map(|_| {
181            (0..secret.len()).map(|_| {
182                *rng ^= *rng << 13; *rng ^= *rng >> 7; *rng ^= *rng << 17;
183                (*rng & 0xff) as u8
184            }).collect()
185        }).collect();
186
187        for i in 0..n {
188            let x = (i + 1) as u8;
189            for j in 0..secret.len() {
190                // P(x) = secret XOR c1*x XOR c2*x^2 XOR ... –≤ GF(256)
191                let mut val = secret[j];
192                let mut xpow: u8 = x;
193                for coeff in &coeffs {
194                    val = gf_add(val, gf_mul(coeff[j], xpow));
195                    xpow = gf_mul(xpow, x);
196                }
197                shards[i][j] = val;
198            }
199        }
200        shards
201    }
202
203    /// –í–æ—Å—Å—Ç–∞–Ω–æ–≤–∏—Ç—å —Å–µ–∫—Ä–µ—Ç –∏–∑ K –æ—Å–∫–æ–ª–∫–æ–≤ ‚Äî GF(256) –∞—Ä–∏—Ñ–º–µ—Ç–∏–∫–∞
204    pub fn reconstruct(shards: &[(u8, Vec<u8>)]) -> Vec<u8> {
205        if shards.is_empty() { return vec![]; }
206        let len = shards[0].1.len();
207        let mut secret = vec![0u8; len];
208
209        for j in 0..len {
210            let mut val: u8 = 0;
211            for (i, (xi, yi)) in shards.iter().enumerate() {
212                let mut num: u8 = yi[j];
213                let mut den: u8 = 1;
214                for (k, (xk, _)) in shards.iter().enumerate() {
215                    if i != k {
216                        num = gf_mul(num, *xk);
217                        den = gf_mul(den, gf_add(*xi, *xk));
218                    }
219                }
220                val = gf_add(val, gf_mul(num, gf_inv(den)));
221            }
222            secret[j] = val;
223        }
224        secret
225    }
226}

252        if shard.is_decoy { self.total_decoys += 1; }
253        else { self.total_shards += 1; }
254        self.nodes.entry(ghost_id.to_string())
255            .or_default().push(shard);
256    }

259                           key_commitment: &str) -> Option<&KeyShard> {
260        self.nodes.get(ghost_id)?
261            .iter()
262            .find(|s| s.key_commitment == key_commitment && !s.is_decoy)
263    }

266        self.nodes.values()
267            .flat_map(|shards| shards.iter())
268            .map(|s| s.memory_kb()).sum()
269    }
270
271    pub fn ghost_count(&self) -> usize { self.nodes.len() }
272}

290impl CryptoVault {
291    pub fn new() -> Self {
292        use std::time::{SystemTime, UNIX_EPOCH};
293        let seed = SystemTime::now().duration_since(UNIX_EPOCH)
294            .unwrap().as_nanos() as u64;
295        CryptoVault {
296            hot: HashMap::new(), cold: HashMap::new(),
297            ghost_network: GhostNetwork::new(),
298            shard_index: HashMap::new(),
299            total_entries: 0, total_zk_proofs: 0,
300            rng: seed ^ 0xdeadbeef_cafebabe,
301        }

303
304    fn encrypt(&mut self, data: &[u8]) -> Vec<u8> {
305        // XOR stream cipher (–≤ prod ‚Äî ChaCha20)
306        self.rng ^= self.rng << 13;
307        self.rng ^= self.rng >> 7;
308        self.rng ^= self.rng << 17;
309        let key = self.rng.to_le_bytes();
310        data.iter().enumerate()
311            .map(|(i, &b)| b ^ key[i % 8]).collect()
312    }
313
314    fn now() -> i64 {
315        use std::time::{SystemTime, UNIX_EPOCH};
316        SystemTime::now().duration_since(UNIX_EPOCH)
317            .unwrap().as_millis() as i64
318    }

320    /// –ü–æ–ª–æ–∂–∏—Ç—å –≤ Hot vault
321    pub fn store_hot(&mut self, key_id: &str, owner_id: &str,
322                     payload: &[u8], rep_required: f64) -> ZkProof {
323        let encrypted = self.encrypt(payload);
324        let proof = ZkProof::generate(owner_id, payload, &mut self.rng);
325        let now = Self::now();
326        let entry = VaultEntry {
327            key_id: key_id.to_string(),
328            owner_id: owner_id.to_string(),
329            tier: VaultTier::Hot,
330            encrypted_payload: encrypted,
331            proof: proof.clone(),
332            created_at: now, accessed_at: now,
333            access_count: 0, reputation_required: rep_required,
334        };
335        if self.hot.len() < HOT_VAULT_LIMIT {
336            self.hot.insert(key_id.to_string(), entry);
337            self.total_entries += 1;
338            self.total_zk_proofs += 1;
339        }
340        proof
341    }

366    /// –û—Å–∫–æ–ª–æ—á–Ω–æ–µ —Ö—Ä–∞–Ω–µ–Ω–∏–µ ‚Äî Veteran –∫–ª—é—á –¥—Ä–æ–±–∏—Ç—Å—è –ø–æ Ghost-—É–∑–ª–∞–º
367    pub fn shard_to_ghosts(&mut self, key_id: &str, owner_id: &str,
368                            payload: &[u8], ghost_ids: &[&str],
369                            n: usize, k: usize) -> ShardingResult {
370        let shards = ShamirScheme::split(payload, n, k, &mut self.rng);
371
372        self.rng ^= self.rng << 13;
373        let commitment = format!("commit_{:016x}", self.rng ^ payload.len() as u64);
374        let owner_commit = format!("owner_{:016x}", self.rng
375            ^ owner_id.bytes().fold(0u64, |a, b| a.wrapping_add(b as u64)));
376
377        let mut shard_map = vec![];
378
379        for (i, shard_data) in shards.iter().enumerate() {
380            let ghost_id = ghost_ids[i % ghost_ids.len()];
381
382            // –†–µ–∞–ª—å–Ω—ã–π –æ—Å–∫–æ–ª–æ–∫
383            let shard = KeyShard {
384                shard_id: (i + 1) as u8,
385                total_shards: n as u8,
386                threshold: k as u8,
387                ghost_node_id: ghost_id.to_string(),
388                shard_data: shard_data.clone(),
389                key_commitment: commitment.clone(),
390                owner_commitment: owner_commit.clone(),
391                is_decoy: false,
392            };
393            self.ghost_network.store_shard(ghost_id, shard);
394            shard_map.push((i as u8 + 1, ghost_id.to_string()));
395
396            // –õ–æ–∂–Ω—ã–µ –æ—Å–∫–æ–ª–∫–∏ –¥–ª—è –º–∞—Å–∫–∏—Ä–æ–≤–∫–∏
397            for _ in 0..2 {
398                self.rng ^= self.rng << 13; self.rng ^= self.rng >> 7;
399                let decoy_data: Vec<u8> = (0..shard_data.len())
400                    .map(|_| { self.rng ^= self.rng << 17; (self.rng & 0xff) as u8 }).collect();
401                let decoy = KeyShard {
402                    shard_id: (i + 1) as u8,
403                    total_shards: n as u8,
404                    threshold: k as u8,
405                    ghost_node_id: ghost_id.to_string(),
406                    shard_data: decoy_data,
407                    key_commitment: format!("decoy_{:08x}", self.rng),
408                    owner_commitment: format!("decoy_owner_{:08x}", self.rng),
409                    is_decoy: true,
410                };
411                self.ghost_network.store_shard(ghost_id, decoy);
412            }
413        }
414
415        self.shard_index.insert(key_id.to_string(), shard_map.clone());
416        self.total_entries += 1;
417
418        ShardingResult {
419            key_id: key_id.to_string(),
420            commitment: commitment.clone(),
421            total_shards: n,
422            threshold: k,
423            ghost_nodes: ghost_ids.iter().take(n).map(|s| s.to_string()).collect(),
424            decoy_shards: n * 2,
425        }
426    }

428    /// –ü–æ–ª—É—á–∏—Ç—å –∏–∑ Hot vault —Å ZK –ø—Ä–æ–≤–µ—Ä–∫–æ–π
429    pub fn retrieve_hot(&mut self, key_id: &str,
430                         proof: &ZkProof, owner_rep: f64) -> VaultResult {
431        if !proof.verify() {
432            return VaultResult::denied("ZK proof –∏—Å—Ç—ë–∫");
433        }
434        match self.hot.get_mut(key_id) {
435            None => VaultResult::denied("–ö–ª—é—á –Ω–µ –Ω–∞–π–¥–µ–Ω –≤ Hot vault"),
436            Some(entry) => {
437                if owner_rep < entry.reputation_required {
438                    return VaultResult::denied(&format!(
439                        "–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–∞—è —Ä–µ–ø—É—Ç–∞—Ü–∏—è: {:.1} < {:.1}",
440                        owner_rep, entry.reputation_required));
441                }
442                entry.access_count += 1;
443                entry.accessed_at = Self::now();
444                VaultResult::success(
445                    entry.encrypted_payload.clone(),
446                    VaultTier::Hot, entry.access_count)
447            }
448        }
449    }
