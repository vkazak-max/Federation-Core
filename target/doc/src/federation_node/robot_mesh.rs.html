151impl DroidNode {
152    pub fn best_protocol(&self) -> Option<&RadioProtocol> {
153        // –ü—Ä–∏–æ—Ä–∏—Ç–µ—Ç: WiFiDirect > Thread > Matter > BT5 > BLE > Zigbee > ZWave
154        let priority = [
155            RadioProtocol::WiFiDirect,
156            RadioProtocol::Thread,
157            RadioProtocol::Matter,
158            RadioProtocol::Bluetooth5,
159            RadioProtocol::BluetoothLE,
160            RadioProtocol::Zigbee,
161            RadioProtocol::ZWave,
162        ];
163        for p in &priority {
164            if self.protocols.contains(p) { return Some(
165                self.protocols.iter().find(|x| *x == p).unwrap()); }
166        }
167        None
168    }
169
170    pub fn can_relay(&self, payload_size: usize) -> bool {
171        self.mesh_enabled && self.firmware_patched &&
172        self.protocols.iter().any(|p| p.max_payload() >= payload_size)
173    }
174
175    pub fn signal_strength_to(&self, other: &DroidNode) -> f32 {
176        let dx = self.position_x - other.position_x;
177        let dy = self.position_y - other.position_y;
178        let dist = (dx*dx + dy*dy).sqrt();
179        // RSSI —É–ø—Ä–æ—â—ë–Ω–Ω–æ: -40 dBm –Ω–∞ 1–º, -6 dBm –Ω–∞ —É–¥–≤–æ–µ–Ω–∏–µ
180        let max_range = self.protocols.iter()
181            .map(|p| p.range_m() as f32).fold(0.0f32, f32::max);
182        if dist > max_range { return -100.0; }
183        -40.0 - 20.0 * (dist.max(0.1)).log10()
184    }
185}

191#[derive(Debug, Clone, Serialize, Deserialize)]
192pub struct StealthPacket {
193    pub packet_id: u64,
194    pub cover_type: String,      // –º–∞—Å–∫–∏—Ä–æ–≤–∫–∞ –ø–æ–¥ —Å–ª—É–∂–µ–±–Ω—ã–π —Ç—Ä–∞—Ñ–∏–∫
195    pub cover_data: Vec<u8>,     // –ª–µ–≥–∏—Ç–∏–º–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ (—Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞, –º–∞—Ä—à—Ä—É—Ç...)
196    pub hidden_payload: Vec<u8>, // –¥–∞–Ω–Ω—ã–µ –§–µ–¥–µ—Ä–∞—Ü–∏–∏
197    pub hidden_offset: usize,    // —Å–º–µ—â–µ–Ω–∏–µ –≤ cover_data
198    pub hop_ttl: u8,
199    pub next_hop: Option<String>,
200    pub checksum: u32,
201}
202
203impl StealthPacket {
204    pub fn embed(federation_data: &[u8], droid: &DroidNode,
205                 rng: &mut u64) -> Option<Self> {
206        let proto = droid.best_protocol()?;
207        if federation_data.len() + 16 > proto.max_payload() {
208            return None; // –¥–∞–Ω–Ω—ã–µ –Ω–µ –≤–ª–µ–∑–∞—é—Ç
209        }
210
211        *rng ^= *rng << 13; *rng ^= *rng >> 7; *rng ^= *rng << 17;
212
213        // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –ø—Ä–∞–≤–¥–æ–ø–æ–¥–æ–±–Ω—ã–µ —Å–ª—É–∂–µ–±–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
214        let cover_size = proto.max_payload().min(128);
215        let mut cover_data: Vec<u8> = (0..cover_size)
216            .map(|_| { *rng ^= *rng << 17; (*rng & 0xff) as u8 }).collect();
217
218        // –ü—Ä—è—á–µ–º —Ñ–µ–¥–µ—Ä–∞–ª—å–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ –≤ —Å–µ—Ä–µ–¥–∏–Ω—É cover_data
219        let offset = cover_size / 3;
220        let end = (offset + federation_data.len()).min(cover_size);
221        cover_data[offset..end].copy_from_slice(
222            &federation_data[..end-offset]);
223
224        let checksum = cover_data.iter()
225            .fold(0u32, |a, &b| a.wrapping_add(b as u32));
226
227        Some(StealthPacket {
228            packet_id: *rng,
229            cover_type: droid.droid_type.stealth_cover().to_string(),
230            cover_data, hidden_payload: federation_data.to_vec(),
231            hidden_offset: offset, hop_ttl: MESH_HOP_TTL,
232            next_hop: None, checksum,
233        })
234    }
235
236    pub fn extract(&self) -> Vec<u8> {
237        self.hidden_payload.clone()
238    }
239
240    pub fn total_size(&self) -> usize {
241        self.cover_data.len() + 32 // overhead
242    }
243}

283    pub fn best_relay(&self, payload_size: usize) -> Option<&DroidNode> {
284        self.droids.values()
285            .filter(|d| d.can_relay(payload_size))
286            .max_by(|a, b| {
287                let score_a = a.droid_type.uptime_pct()
288                    * a.best_protocol().map(|p| p.max_payload() as f64).unwrap_or(0.0);
289                let score_b = b.droid_type.uptime_pct()
290                    * b.best_protocol().map(|p| p.max_payload() as f64).unwrap_or(0.0);
291                score_a.partial_cmp(&score_b).unwrap()
292            })
293    }

295    pub fn relay_packet(&mut self, data: &[u8]) -> RelayResult {
296        // –°–æ–±–∏—Ä–∞–µ–º –¥–∞–Ω–Ω—ã–µ –±–µ–∑ borrow –Ω–∞ self
297        let relay_info = self.best_relay(data.len()).map(|droid| {
298            let proto = droid.best_protocol().unwrap();
299            let latency = match proto {
300                RadioProtocol::Bluetooth5  => 5,
301                RadioProtocol::BluetoothLE => 15,
302                RadioProtocol::Zigbee      => 30,
303                RadioProtocol::Thread      => 20,
304                RadioProtocol::WiFiDirect  => 2,
305                _                          => 25,
306            };
307            (droid.droid_id.clone(), proto.name().to_string(),
308             latency, droid.droid_type.stealth_cover().to_string())
309        });
310
311        match relay_info {
312            None => RelayResult {
313                success: false, droid_id: "none".into(),
314                protocol: "none".into(), latency_ms: 0,
315                stealth_cover: "none".into(),
316                reason: "–Ω–µ—Ç –ø–æ–¥—Ö–æ–¥—è—â–µ–≥–æ –¥—Ä–æ–∏–¥–∞".into(),
317            },
318            Some((droid_id, protocol, latency_ms, stealth_cover)) => {
319                self.packets_relayed += 1;
320                self.bytes_relayed += data.len() as u64;
321                RelayResult {
322                    success: true, droid_id, protocol,
323                    latency_ms, stealth_cover, reason: "OK".into(),
324                }
325            }
326        }
327    }

329    pub fn mesh_coverage(&self) -> f64 {
330        // –ü—Ä–æ—Ü–µ–Ω—Ç –ø–ª–æ—â–∞–¥–∏ –∫–≤–∞—Ä—Ç–∏—Ä—ã –ø–æ–∫—Ä—ã—Ç–æ–π –º–µ—à-—Å–µ—Ç—å—é
331        let active = self.droids.values()
332            .filter(|d| d.mesh_enabled).count();
333        (active as f64 / self.droids.len().max(1) as f64).min(1.0)
334    }

336    pub fn bastion_stats(&self) -> BastionStats {
337        let active_droids = self.droids.values()
338            .filter(|d| d.mesh_enabled && d.firmware_patched).count();
339        let total_uptime: f64 = self.droids.values()
340            .map(|d| d.droid_type.uptime_pct()).sum::<f64>()
341            / self.droids.len().max(1) as f64;
342
343        BastionStats {
344            apartment_id: self.apartment_id.clone(),
345            total_droids: self.droids.len(),
346            active_droids,
347            mesh_coverage: self.mesh_coverage(),
348            avg_uptime: total_uptime,
349            packets_relayed: self.packets_relayed,
350            bytes_relayed: self.bytes_relayed,

409    pub fn route_through_mesh(&mut self, from: &str, to: &str,
410                               _data: &[u8]) -> MeshRouteResult {
411        // BFS –ø–æ –±–∞—Å—Ç–∏–æ–Ω–∞–º
412        let mut visited = std::collections::HashSet::new();
413        let mut queue = std::collections::VecDeque::new();
414        queue.push_back((from.to_string(), vec![from.to_string()]));
415        visited.insert(from.to_string());
416
417        while let Some((current, path)) = queue.pop_front() {
418            if current == to {
419                self.total_relayed += 1;
420                return MeshRouteResult {
421                    success: true, hops: path.len() as u8 - 1,
422                    path: path.clone(), latency_ms: path.len() as u32 * 15,
423                    reason: "route_found".into(),
424                };
425            }
426            if path.len() >= MESH_HOP_TTL as usize { continue; }
427
428            let neighbors = self.bastions.get(&current)
429                .map(|b| b.neighbors.clone()).unwrap_or_default();
430            for neighbor in neighbors {
431                if !visited.contains(&neighbor) {
432                    visited.insert(neighbor.clone());
433                    let mut new_path = path.clone();
434                    new_path.push(neighbor.clone());
435                    queue.push_back((neighbor, new_path));
436                }
437            }
438        }
439        MeshRouteResult {
440            success: false, hops: 0, path: vec![],
441            latency_ms: 0, reason: "no_route".into(),
442        }
443    }
444
445    pub fn city_stats(&self) -> CityStats {
446        let total_droids: usize = self.bastions.values()
447            .map(|b| b.droids.len()).sum();
448        let active_bastions = self.bastions.values()
449            .filter(|b| b.mesh_active).count();
450        CityStats {
451            city: self.city.clone(),
452            total_bastions: self.bastions.len(),
453            active_bastions, total_droids,
454            total_relayed: self.total_relayed,
455        }
456    }
457}
