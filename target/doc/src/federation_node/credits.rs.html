39    pub has_ml_censor: bool,      // –µ—Å—Ç—å ML-—Ü–µ–Ω–∑—É—Ä–∞?

44impl RegionDifficulty {
45    pub fn new(code: &str, block_rate: f64, dpi_level: u8,
46               has_ml: bool, shutdown: bool) -> Self {
47        let mut r = RegionDifficulty {
48            region_code: code.to_string(),
49            block_rate, dpi_level, has_ml_censor: has_ml,
50            internet_shutdown: shutdown, difficulty_score: 0.0,
51        };
52        r.difficulty_score = r.compute_difficulty();
53        r
54    }
55
56    pub fn compute_difficulty(&self) -> f64 {
57        let base = self.block_rate;
58        let dpi_factor = self.dpi_level as f64 / 5.0 * 0.3;
59        let ml_factor = if self.has_ml_censor { 0.2 } else { 0.0 };
60        let shutdown_factor = if self.internet_shutdown { 0.3 } else { 0.0 };
61        (base + dpi_factor + ml_factor + shutdown_factor).min(1.0)
62    }
63
64    pub fn difficulty_multiplier(&self) -> f64 {
65        1.0 + self.difficulty_score * DIFFICULTY_SCALE
66    }
67
68    pub fn label(&self) -> &str {
69        match self.difficulty_score as u8 {
70            _ if self.difficulty_score >= 0.8 => "üî¥ EXTREME",
71            _ if self.difficulty_score >= 0.6 => "üü† HIGH",
72            _ if self.difficulty_score >= 0.4 => "üü° MEDIUM",
73            _ if self.difficulty_score >= 0.2 => "üü¢ LOW",
74            _                                  => "‚ö™ MINIMAL",
75        }
76    }
77}

116impl BypassEvent {
117    pub fn compute_credits(tactic: &str, packets: u64,
118        difficulty_mult: f64, cpu_drained: f64,
119        has_evidence: bool) -> f64 {
120
121        let tactic_coeff = match tactic {
122            "AikiReflection"   => AIKI_BONUS,
123            "CumulativeStrike" => STRIKE_BONUS,
124            "StandoffDecoy"    => DECOY_BONUS,
125            "Hybrid"           => (AIKI_BONUS + STRIKE_BONUS) / 2.0,
126            _                  => PASSIVE_BONUS,
127        };
128
129        let base = BYPASS_BASE * difficulty_mult;
130        let tactic_bonus = base * tactic_coeff;
131        let packet_bonus = (packets as f64).ln().max(1.0) * 0.1;
132        let cpu_bonus = cpu_drained * 0.5; // –±–æ–Ω—É—Å –∑–∞ –∏—Å—Ç–æ—â–µ–Ω–∏–µ —Ü–µ–Ω–∑–æ—Ä–∞
133        let evidence_bonus = if has_evidence { EVIDENCE_BONUS } else { 0.0 };
134
135        base + tactic_bonus + packet_bonus + cpu_bonus + evidence_bonus
136    }
137}

143#[derive(Debug, Default, Serialize, Deserialize)]
144pub struct CreditLedger {
145    pub balances: HashMap<String, f64>,
146    pub events: Vec<BypassEvent>,
147    pub streaks: HashMap<String, u32>,      // —Å–µ—Ä–∏—è —É—Å–ø–µ—Ö–æ–≤ —É–∑–ª–∞
148    pub total_credits_issued: f64,
149    pub event_counter: u64,
150}
151
152impl CreditLedger {
153    pub fn new() -> Self { Self::default() }
154
155    pub fn record_bypass(&mut self, node_id: &str, region: &str,
156        tactic: &str, packets: u64, cpu_drained: f64,
157        difficulty: &RegionDifficulty, has_evidence: bool) -> f64 {
158
159        use std::time::{SystemTime, UNIX_EPOCH};
160        let now = SystemTime::now().duration_since(UNIX_EPOCH)
161            .unwrap().as_millis() as i64;
162
163        // Streak multiplier
164        let streak = self.streaks.entry(node_id.to_string()).or_insert(0);
165        *streak += 1;
166        let streak_mult = (1.0 + *streak as f64 * STREAK_STEP)
167            .min(MAX_STREAK_MULT);
168
169        let base_credits = BypassEvent::compute_credits(
170            tactic, packets,
171            difficulty.difficulty_multiplier(),
172            cpu_drained, has_evidence,
173        );
174        let total = base_credits * streak_mult;
175
176        // –û–±–Ω–æ–≤–ª—è–µ–º –±–∞–ª–∞–Ω—Å
177        *self.balances.entry(node_id.to_string()).or_insert(0.0) += total;
178        self.total_credits_issued += total;
179        self.event_counter += 1;
180
181        let event = BypassEvent {
182            event_id: self.event_counter,
183            node_id: node_id.to_string(),
184            region: region.to_string(),
185            tactic: tactic.to_string(),
186            packets_delivered: packets,
187            censor_cpu_drained: cpu_drained,
188            difficulty: difficulty.difficulty_score,
189            has_evidence,
190            credits_earned: total,
191            timestamp: now,
192        };
193        self.events.push(event);
194        total
195    }
196
197    pub fn record_failure(&mut self, node_id: &str) {
198        // –ü—Ä–æ–≤–∞–ª —Å–±—Ä–∞—Å—ã–≤–∞–µ—Ç —Å–µ—Ä–∏—é
199        if let Some(s) = self.streaks.get_mut(node_id) { *s = 0; }
200    }
201
202    pub fn balance(&self, node_id: &str) -> f64 {
203        self.balances.get(node_id).cloned().unwrap_or(0.0)
204    }
205
206    pub fn top_nodes(&self, n: usize) -> Vec<(String, f64)> {
207        let mut v: Vec<_> = self.balances.iter()
208            .map(|(k, v)| (k.clone(), *v)).collect();
209        v.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
210        v.into_iter().take(n).collect()
211    }
212
213    pub fn stats(&self) -> LedgerStats {
214        let avg = if !self.balances.is_empty() {
215            self.balances.values().sum::<f64>() / self.balances.len() as f64
216        } else { 0.0 };
217
218        let max_streak = self.streaks.values().cloned().max().unwrap_or(0);
219        let top = self.top_nodes(1);
220
221        LedgerStats {
222            total_nodes: self.balances.len(),
223            total_events: self.events.len(),
224            total_issued: self.total_credits_issued,
225            avg_balance: avg,
226            max_streak,
227            top_node: top.into_iter().next(),
228        }
229    }
230}

328impl EcoProfile {
329    pub fn new(node_id: &str, hw_age_years: u32, is_recycled: bool) -> Self {
330        let hw_age = HardwareAge::from_years(hw_age_years);
331        let mut recycle_mult = hw_age.recycle_multiplier();
332        if is_recycled { recycle_mult = (recycle_mult * 1.2).min(MAX_RECYCLE_MULT); }
333        EcoProfile {
334            node_id: node_id.to_string(),
335            hw_age_years, hw_age, recycle_mult,
336            total_eco_earned: 0.0, upgrade_fund_paid: 0.0,
337            is_recycled_device: is_recycled,
338        }
339    }
340
341    pub fn apply(&mut self, base_credits: f64) -> EcoReward {
342        let recycled_credits = base_credits * self.recycle_mult;
343        let eco_bonus        = recycled_credits - base_credits;
344        let upgrade_contrib  = recycled_credits * UPGRADE_FUND_RATE;
345        let net_credits      = recycled_credits - upgrade_contrib;
346
347        self.total_eco_earned   += eco_bonus;
348        self.upgrade_fund_paid  += upgrade_contrib;
349
350        EcoReward {
351            base_credits,
352            recycle_mult: self.recycle_mult,
353            recycled_credits,
354            eco_bonus,
355            upgrade_fund_contribution: upgrade_contrib,
356            net_credits,
357            hw_age_label: self.hw_age.label().to_string(),
358        }
359    }
360}

385impl UpgradeFund {

401    pub fn top_contributors(&self, n: usize) -> Vec<(&str, f64)> {
402        let mut map: HashMap<&str, f64> = HashMap::new();
403        for (node, amt) in &self.contributions {
404            *map.entry(node.as_str()).or_insert(0.0) += amt;
405        }
406        let mut v: Vec<(&str, f64)> = map.into_iter().collect();
407        v.sort_by(|a,b| b.1.partial_cmp(&a.1).unwrap());
408        v.into_iter().take(n).collect()
409    }
