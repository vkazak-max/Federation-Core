18pub const REP_BASE_DELIVERY: f64    = 0.10;  // –∑–∞ –∫–∞–∂–¥—É—é –¥–æ—Å—Ç–∞–≤–∫—É
19pub const REP_AIKI_MULT: f64        = 3.0;   // –º–Ω–æ–∂–∏—Ç–µ–ª—å AikiReflection
20pub const REP_STRIKE_MULT: f64      = 2.0;   // –º–Ω–æ–∂–∏—Ç–µ–ª—å CumulativeStrike
21pub const REP_DECOY_MULT: f64       = 1.5;   // –º–Ω–æ–∂–∏—Ç–µ–ª—å StandoffDecoy
22pub const REP_BETRAYAL_SLASH: f64   = 0.50;  // -50% —Ä–µ–ø—É—Ç–∞—Ü–∏–∏ –∑–∞ –ø—Ä–µ–¥–∞—Ç–µ–ª—å—Å—Ç–≤–æ
23pub const REP_FAILURE_DECAY: f64    = 0.02;  // -2% –∑–∞ –∫–∞–∂–¥—ã–π –ø—Ä–æ–≤–∞–ª
24pub const DAO_WEIGHT_EXPONENT: f64  = 0.7;   // —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ –¥–ª—è DAO
25pub const LEGEND_THRESHOLD: f64     = 500.0;
26pub const VETERAN_THRESHOLD: f64    = 100.0;
27pub const TRUSTED_THRESHOLD: f64    = 30.0;
28pub const RELIABLE_THRESHOLD: f64   = 10.0;
29pub const NEWCOMER_THRESHOLD: f64   = 1.0;

35#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
36pub enum ReputationTier {
37    Ghost,      // < 1.0   ‚Äî –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω—ã–π
38    Newcomer,   // 1-10    ‚Äî –Ω–æ–≤–∏—á–æ–∫
39    Reliable,   // 10-30   ‚Äî –Ω–∞–¥—ë–∂–Ω—ã–π
40    Trusted,    // 30-100  ‚Äî –¥–æ–≤–µ—Ä–µ–Ω–Ω—ã–π
41    Veteran,    // 100-500 ‚Äî –≤–µ—Ç–µ—Ä–∞–Ω
42    Legend,     // 500+    ‚Äî –ª–µ–≥–µ–Ω–¥–∞
43}
44
45impl ReputationTier {
46    pub fn from_score(score: f64) -> Self {
47        if score >= LEGEND_THRESHOLD  { ReputationTier::Legend   }
48        else if score >= VETERAN_THRESHOLD  { ReputationTier::Veteran  }
49        else if score >= TRUSTED_THRESHOLD  { ReputationTier::Trusted  }
50        else if score >= RELIABLE_THRESHOLD { ReputationTier::Reliable }
51        else if score >= NEWCOMER_THRESHOLD { ReputationTier::Newcomer }
52        else                                { ReputationTier::Ghost    }
53    }
54
55    pub fn name(&self) -> &str {
56        match self {
57            ReputationTier::Ghost    => "üëª Ghost",
58            ReputationTier::Newcomer => "üå± Newcomer",
59            ReputationTier::Reliable => "üîµ Reliable",
60            ReputationTier::Trusted  => "üü£ Trusted",
61            ReputationTier::Veteran  => "üü† Veteran",
62            ReputationTier::Legend   => "üåü Legend",
63        }
64    }
65
66    pub fn dao_weight_bonus(&self) -> f64 {
67        match self {
68            ReputationTier::Ghost    => 0.0,
69            ReputationTier::Newcomer => 0.5,
70            ReputationTier::Reliable => 1.0,
71            ReputationTier::Trusted  => 1.5,
72            ReputationTier::Veteran  => 2.0,
73            ReputationTier::Legend   => 3.0,
74        }
75    }
76}

123impl NodeReputation {
124    pub fn new(node_id: &str) -> Self {
125        NodeReputation {
126            node_id: node_id.to_string(),
127            score: 0.0,
128            tier: ReputationTier::Ghost,
129            total_deliveries: 0,
130            successful_deliveries: 0,
131            aiki_victories: 0,
132            betrayals: 0,
133            ethics_violations: 0,
134            uptime_days: 0,
135            dao_participations: 0,
136            history: vec![],
137            is_blacklisted: false,
138            stake: 0.0,
139        }
140    }
141
142    pub fn delivery_rate(&self) -> f64 {
143        if self.total_deliveries == 0 { return 0.0; }
144        self.successful_deliveries as f64 / self.total_deliveries as f64
145    }
146
147    pub fn dao_voting_weight(&self) -> f64 {
148        if self.is_blacklisted { return 0.0; }
149        // DAO –≤–µ—Å = score^0.7 + tier_bonus
150        // –ù–µ–ª—å–∑—è –∫—É–ø–∏—Ç—å ‚Äî —Ç–æ–ª—å–∫–æ –∑–∞—Ä–∞–±–æ—Ç–∞—Ç—å
151        let base = self.score.powf(DAO_WEIGHT_EXPONENT);
152        base + self.tier.dao_weight_bonus()
153    }
154
155    pub fn update_tier(&mut self) {
156        self.tier = ReputationTier::from_score(self.score);
157    }
158}

171impl ReputationRegistry {
172    pub fn new() -> Self {
173        ReputationRegistry {
174            nodes: HashMap::new(),

176            total_slashes: 0,

181    fn get_or_create(&mut self, node_id: &str) -> &mut NodeReputation {
182        self.nodes.entry(node_id.to_string())
183            .or_insert_with(|| NodeReputation::new(node_id))
184    }

186    fn now() -> i64 {
187        use std::time::{SystemTime, UNIX_EPOCH};
188        SystemTime::now().duration_since(UNIX_EPOCH)
189            .unwrap().as_millis() as i64
190    }

192    /// –£—Å–ø–µ—à–Ω–∞—è –¥–æ—Å—Ç–∞–≤–∫–∞
193    pub fn record_delivery(&mut self, node_id: &str, tactic: &str,
194                            region_difficulty: f64) -> f64 {
195        let tactic_mult = match tactic {
196            "AikiReflection"   => REP_AIKI_MULT,
197            "CumulativeStrike" => REP_STRIKE_MULT,
198            "StandoffDecoy"    => REP_DECOY_MULT,
199            "Hybrid"           => (REP_AIKI_MULT + REP_STRIKE_MULT) / 2.0,
200            _                  => 1.0,
201        };
202        let delta = REP_BASE_DELIVERY * (1.0 + region_difficulty * 3.0) * tactic_mult;
203
204        let node = self.get_or_create(node_id);
205        if node.is_blacklisted { return 0.0; }
206        node.score += delta;
207        node.stake += delta * 0.1;
208        node.total_deliveries += 1;
209        node.successful_deliveries += 1;
210        node.update_tier();
211
212        let event = ReputationEvent {
213            node_id: node_id.to_string(),
214            kind: ReputationEventKind::SuccessfulDelivery {
215                tactic: tactic.to_string(),
216                region_difficulty,
217            },
218            rep_delta: delta, timestamp: Self::now(), is_slash: false,
219        };
220        node.history.push(event);
221        self.total_events += 1;
222        delta
223    }

225    /// –ü—Ä–æ–≤–∞–ª –¥–æ—Å—Ç–∞–≤–∫–∏
226    pub fn record_failure(&mut self, node_id: &str, region: &str) -> f64 {
227        let node = self.get_or_create(node_id);
228        let delta = -(node.score * REP_FAILURE_DECAY).max(0.01);
229        node.score = (node.score + delta).max(0.0);
230        node.total_deliveries += 1;
231        node.stake = (node.stake + delta * 0.05).max(0.0);
232        node.update_tier();
233        let event = ReputationEvent {
234            node_id: node_id.to_string(),
235            kind: ReputationEventKind::FailedDelivery {
236                region: region.to_string() },
237            rep_delta: delta, timestamp: Self::now(), is_slash: false,
238        };
239        node.history.push(event);
240        self.total_events += 1;
241        delta
242    }

245    pub fn record_aiki_victory(&mut self, node_id: &str,

248        let node = self.get_or_create(node_id);
249        if node.is_blacklisted { return 0.0; }
250        node.score += delta;
251        node.stake += delta * 0.2;
252        node.aiki_victories += 1;
253        node.update_tier();
254        let event = ReputationEvent {
255            node_id: node_id.to_string(),
256            kind: ReputationEventKind::AikiVictory { censor_cpu_drained: cpu_drained },
257            rep_delta: delta, timestamp: Self::now(), is_slash: false,
258        };

264    /// –ù–∞—Ä—É—à–µ–Ω–∏–µ —ç—Ç–∏–∫–∏
265    pub fn record_ethics_violation(&mut self, node_id: &str,
266                                    violation: &str, severity: f64) -> f64 {
267        let node = self.get_or_create(node_id);
268        let delta = -(node.score * severity * 0.3).max(0.1);
269        node.score = (node.score + delta).max(0.0);
270        node.ethics_violations += 1;
271        node.update_tier();
272        let event = ReputationEvent {
273            node_id: node_id.to_string(),
274            kind: ReputationEventKind::EthicsViolation {
275                violation: violation.to_string(), severity },
276            rep_delta: delta, timestamp: Self::now(), is_slash: true,
277        };
278        node.history.push(event);
279        self.total_events += 1;
280        self.total_slashes += 1;
281        delta
282    }

284    /// –ü–†–ï–î–ê–¢–ï–õ–¨–°–¢–í–û ‚Äî –ø–æ–º–æ—â—å —Ü–µ–Ω–∑–æ—Ä—É
285    pub fn record_betrayal(&mut self, node_id: &str,
286                           evidence_hash: &str) -> f64 {
287        // –í—ã—á–∏—Å–ª—è–µ–º –≤—Å—ë –¥–æ –∏–∑–º–µ–Ω–µ–Ω–∏—è self
288        let (slash, newly_blacklisted) = {
289            let node = self.get_or_create(node_id);
290            let slash = node.score * REP_BETRAYAL_SLASH;
291            node.score -= slash;
292            node.stake = 0.0;
293            node.betrayals += 1;
294            let newly_blacklisted = if node.betrayals >= 3 {
295                node.is_blacklisted = true;
296                node.score = 0.0;
297                true
298            } else { false };
299            node.update_tier();
300            let event = ReputationEvent {
301                node_id: node_id.to_string(),
302                kind: ReputationEventKind::Betrayal {
303                    evidence_hash: evidence_hash.to_string() },
304                rep_delta: -slash, timestamp: Self::now(), is_slash: true,
305            };
306            node.history.push(event);
307            (slash, newly_blacklisted)
308        };
309        // –¢–µ–ø–µ—Ä—å borrow –∑–∞–∫—Ä—ã—Ç ‚Äî –º–æ–∂–Ω–æ –æ–±–Ω–æ–≤–∏—Ç—å —Å—á—ë—Ç—á–∏–∫–∏
310        if newly_blacklisted { self.blacklisted_count += 1; }
311        self.total_events += 1;
312        self.total_slashes += 1;
313        -slash
314    }

325            node_id: node_id.to_string(),
326            kind: ReputationEventKind::DaoParticipation {
327                proposal_id: proposal_id.to_string() },

335    /// –î–æ–ª–≥–æ—Å—Ä–æ—á–Ω—ã–π –∞–ø—Ç–∞–π–º
336    pub fn record_uptime(&mut self, node_id: &str, days: u32) -> f64 {
337        let node = self.get_or_create(node_id);
338        let delta = (days as f64).sqrt() * 0.5;
339        node.score += delta;
340        node.uptime_days += days;
341        node.update_tier();
342        let event = ReputationEvent {
343            node_id: node_id.to_string(),
344            kind: ReputationEventKind::LongTermUptime { days },
345            rep_delta: delta, timestamp: Self::now(), is_slash: false,
346        };
347        node.history.push(event);
348        self.total_events += 1;
349        delta
350    }

352    pub fn leaderboard(&self, n: usize) -> Vec<(&NodeReputation, usize)> {
353        let mut v: Vec<&NodeReputation> = self.nodes.values()
354            .filter(|n| !n.is_blacklisted).collect();
355        v.sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap());
356        v.into_iter().take(n).enumerate()
357            .map(|(i, n)| (n, i + 1)).collect()
358    }
359
360    pub fn dao_weights(&self) -> Vec<(String, f64)> {
361        let mut v: Vec<(String, f64)> = self.nodes.values()
362            .map(|n| (n.node_id.clone(), n.dao_voting_weight()))
363            .collect();
364        v.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
365        v
366    }

368    pub fn stats(&self) -> RegistryStats {
369        let scores: Vec<f64> = self.nodes.values()
370            .map(|n| n.score).collect();
371        let avg = if scores.is_empty() { 0.0 }
372            else { scores.iter().sum::<f64>() / scores.len() as f64 };
373        RegistryStats {
374            total_nodes: self.nodes.len(),
375            total_events: self.total_events,
376            total_slashes: self.total_slashes,
377            blacklisted: self.blacklisted_count,
378            avg_score: avg,
379            legends: self.nodes.values()
380                .filter(|n| n.tier == ReputationTier::Legend).count(),
381            veterans: self.nodes.values()
382                .filter(|n| n.tier == ReputationTier::Veteran).count(),
383        }
384    }

437#[derive(Debug, Clone, Serialize, Deserialize)]
438pub struct TrustEdge {
439    pub from: String,
440    pub to: String,
441    pub weight: f64,       // 0.0-1.0
442    pub vouches: u32,      // —Å–∫–æ–ª—å–∫–æ —Ä–∞–∑ –ø–æ—Ä—É—á–∏–ª–∏—Å—å
443    pub betrayals: u32,    // —Å–∫–æ–ª—å–∫–æ —Ä–∞–∑ –ø—Ä–µ–¥–∞–ª–∏
444    pub created_at: i64,
445}
446
447impl TrustEdge {
448    pub fn new(from: &str, to: &str, initial_weight: f64) -> Self {
449        TrustEdge {
450            from: from.to_string(), to: to.to_string(),
451            weight: initial_weight.clamp(0.0, 1.0),
452            vouches: 1, betrayals: 0,
453            created_at: 0,
454        }
455    }
456    pub fn effective_weight(&self) -> f64 {
457        if self.betrayals > 0 {
458            (self.weight * 0.5f64.powi(self.betrayals as i32)).max(0.0)
459        } else {
460            self.weight
461        }
462    }
463    pub fn vouch(&mut self) {
464        self.vouches += 1;
465        self.weight = (self.weight + 0.05).min(1.0);
466    }
467    pub fn betray(&mut self) {
468        self.betrayals += 1;
469        self.weight = (self.weight * 0.5).max(0.0);
470    }
471}

474// TrustGraph ‚Äî –≥—Ä–∞—Ñ –¥–æ–≤–µ—Ä–∏—è

476
477pub struct TrustGraph {
478    pub edges: Vec<TrustEdge>,
479    pub trust_ranks: HashMap<String, f64>,
480    pub iterations_run: u32,
481}
482
483impl TrustGraph {
484    pub fn new() -> Self {
485        TrustGraph { edges: vec![], trust_ranks: HashMap::new(), iterations_run: 0 }
486    }
487
488    pub fn add_edge(&mut self, from: &str, to: &str, weight: f64) {
489        if let Some(e) = self.edges.iter_mut()
490            .find(|e| e.from == from && e.to == to) {
491            e.weight = weight.clamp(0.0, 1.0);
492        } else {
493            self.edges.push(TrustEdge::new(from, to, weight));
494        }
495    }
496
497    pub fn vouch(&mut self, from: &str, to: &str) {
498        if let Some(e) = self.edges.iter_mut()
499            .find(|e| e.from == from && e.to == to) {
500            e.vouch();
501        } else {
502            self.edges.push(TrustEdge::new(from, to, 0.5));
503        }
504    }
505
506    pub fn betray(&mut self, from: &str, to: &str) {
507        if let Some(e) = self.edges.iter_mut()
508            .find(|e| e.from == from && e.to == to) {
509            e.betray();
510        }
511    }
512
513    // –í—Å–µ —É–∑–ª—ã –≤ –≥—Ä–∞—Ñ–µ
514    fn all_nodes(&self) -> Vec<String> {
515        let mut nodes = std::collections::HashSet::new();
516        for e in &self.edges {
517            nodes.insert(e.from.clone());
518            nodes.insert(e.to.clone());
519        }
520        nodes.into_iter().collect()
521    }
522
523    // –ò—Å—Ö–æ–¥—è—â–∏–µ —Ä—ë–±—Ä–∞ –æ—Ç —É–∑–ª–∞
524    fn outgoing(&self, node: &str) -> Vec<&TrustEdge> {
525        self.edges.iter().filter(|e| e.from == node).collect()
526    }
527
528    // PageRank-–ø–æ–¥–æ–±–Ω—ã–π –∞–ª–≥–æ—Ä–∏—Ç–º –¥–ª—è –¥–æ–≤–µ—Ä–∏—è
529    pub fn compute_trust_rank(&mut self, seed_reputations: &HashMap<String, f64>) {
530        let nodes = self.all_nodes();
531        let n = nodes.len().max(1) as f64;
532
533        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è: –±–∞–∑–æ–≤–æ–µ –¥–æ–≤–µ—Ä–∏–µ –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ —Ä–µ–ø—É—Ç–∞—Ü–∏–∏
534        let total_rep: f64 = seed_reputations.values().sum::<f64>().max(1.0);
535        let mut ranks: HashMap<String, f64> = nodes.iter().map(|node| {
536            let base = seed_reputations.get(node).copied().unwrap_or(1.0);
537            (node.clone(), base / total_rep)
538        }).collect();
539
540        // PageRank –∏—Ç–µ—Ä–∞—Ü–∏–∏
541        for _ in 0..PAGERANK_ITERATIONS {
542            let mut new_ranks: HashMap<String, f64> = nodes.iter()
543                .map(|nd| (nd.clone(), (1.0 - PAGERANK_DAMPING) / n)).collect();
544
545            for node in &nodes {
546                let outgoing = self.outgoing(node);
547                let total_weight: f64 = outgoing.iter()
548                    .map(|e| e.effective_weight()).sum::<f64>().max(1e-9);
549                let rank = ranks.get(node).copied().unwrap_or(0.0);
550
551                for edge in outgoing {
552                    let contribution = PAGERANK_DAMPING * rank
553                        * (edge.effective_weight() / total_weight);
554                    *new_ranks.entry(edge.to.clone()).or_insert(0.0) += contribution;
555                }
556            }
557            ranks = new_ranks;
558        }
559
560        // –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º 0..1
561        let max_rank = ranks.values().cloned().fold(0.0f64, f64::max).max(1e-9);
562        self.trust_ranks = ranks.into_iter()
563            .map(|(k,v)| (k, v / max_rank)).collect();
564        self.iterations_run = PAGERANK_ITERATIONS;
565    }
566
567    pub fn trust_rank_of(&self, node: &str) -> f64 {
568        self.trust_ranks.get(node).copied().unwrap_or(0.0)
569    }
570
571    // –¢—Ä–∞–Ω–∑–∏—Ç–∏–≤–Ω–æ–µ –¥–æ–≤–µ—Ä–∏–µ –æ—Ç A –¥–æ B —á–µ—Ä–µ–∑ –≥—Ä–∞—Ñ (BFS —Å –∑–∞—Ç—É—Ö–∞–Ω–∏–µ–º)
572    pub fn transitive_trust(&self, from: &str, to: &str) -> f64 {
573        if from == to { return 1.0; }
574        let mut visited = std::collections::HashSet::new();
575        let mut queue = std::collections::VecDeque::new();
576        queue.push_back((from.to_string(), 1.0f64));
577        visited.insert(from.to_string());
578        let mut best = 0.0f64;
579
580        while let Some((node, trust)) = queue.pop_front() {
581            if trust < MIN_TRUST_EDGE { continue; }
582            for edge in self.outgoing(&node) {
583                let new_trust = trust * edge.effective_weight() * (1.0 - TRUST_DECAY);
584                if edge.to == to { best = best.max(new_trust); continue; }
585                if !visited.contains(&edge.to) && new_trust >= MIN_TRUST_EDGE {
586                    visited.insert(edge.to.clone());
587                    queue.push_back((edge.to.clone(), new_trust));
588                }
589            }
590        }
591        best
592    }
593
594    pub fn top_trusted(&self, n: usize) -> Vec<(&str, f64)> {
595        let mut v: Vec<(&str, f64)> = self.trust_ranks.iter()
596            .map(|(k,v)| (k.as_str(), *v)).collect();
597        v.sort_by(|a,b| b.1.partial_cmp(&a.1).unwrap());
598        v.into_iter().take(n).collect()
599    }
600
601    pub fn stats(&self) -> String {
602        format!("nodes={}  edges={}  iterations={}",
603            self.all_nodes().len(), self.edges.len(), self.iterations_run)
604    }
605}
