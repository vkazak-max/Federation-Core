21#[derive(Debug, Clone, Serialize, Deserialize)]
22pub struct LatencyDistribution {
23    pub mean: f64,
24    pub std_dev: f64,
25    pub samples: Vec<f64>,
26}
27
28impl LatencyDistribution {
29    pub fn new(mean: f64, std_dev: f64) -> Self {
30        Self { mean, std_dev, samples: Vec::new() }
31    }
32
33    pub fn add_sample(&mut self, sample_ms: f64) {
34        const MAX_SAMPLES: usize = 50;
35        self.samples.push(sample_ms);
36        if self.samples.len() > MAX_SAMPLES {
37            self.samples.remove(0);
38        }
39        self.recalculate();
40    }
41
42    fn recalculate(&mut self) {
43        if self.samples.is_empty() { return; }
44        let n = self.samples.len() as f64;
45        self.mean = self.samples.iter().sum::<f64>() / n;
46        let variance = self.samples.iter().map(|x| (x - self.mean).powi(2)).sum::<f64>() / n;
47        self.std_dev = variance.sqrt();
48    }
49}
50
51impl SsauTensor {
52    pub fn new(from_node: &str, to_node: &str, latency_ms: f64, bandwidth_mbps: f64) -> Self {
53        use std::time::{SystemTime, UNIX_EPOCH};
54        let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_millis() as i64;
55        Self {
56            from_node: from_node.to_string(),
57            to_node: to_node.to_string(),
58            latency: LatencyDistribution::new(latency_ms, 0.0),
59            jitter: 0.0,
60            bandwidth: bandwidth_mbps,
61            reliability: 1.0,
62            energy_cost: 1.0,
63            updated_at: now,
64            version: 1,
65        }
66    }
67
68    pub fn to_vector(&self) -> [f64; SSAU_DIMENSIONS] {
69        [self.latency.mean, self.jitter, self.bandwidth, self.reliability, self.energy_cost]
70    }
71
72    pub fn update_measurement(&mut self, latency_ms: f64, bandwidth_mbps: f64) {
73        use std::time::{SystemTime, UNIX_EPOCH};
74        let old_mean = self.latency.mean;
75        self.latency.add_sample(latency_ms);
76        self.jitter = (self.latency.mean - old_mean).abs();
77        self.bandwidth = bandwidth_mbps;
78        self.updated_at = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_millis() as i64;
79        self.version += 1;
80    }
81
82    pub fn is_fresh(&self, max_age_ms: i64) -> bool {
83        use std::time::{SystemTime, UNIX_EPOCH};
84        let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_millis() as i64;
85        (now - self.updated_at) < max_age_ms
86    }
87}

90pub struct RouteHealthScore {
91    pub entropy: f64,
92    pub stability_score: f64,
93    pub is_unstable: bool,
94    pub total_latency_ms: f64,
95    pub min_bandwidth_mbps: f64,
96}
97
98pub fn shannon_entropy(tensors: &[&SsauTensor]) -> RouteHealthScore {
99    if tensors.is_empty() {
100        return RouteHealthScore {
101            entropy: f64::INFINITY,
102            stability_score: 0.0,
103            is_unstable: true,
104            total_latency_ms: f64::INFINITY,
105            min_bandwidth_mbps: 0.0,
106        };
107    }
108    let mut entropy = 0.0_f64;
109    let mut total_latency = 0.0_f64;
110    let mut min_bandwidth = f64::INFINITY;
111    for tensor in tensors {
112        let p = tensor.reliability.clamp(1e-10, 1.0 - 1e-10);
113        entropy += -(p * p.log2());
114        total_latency += tensor.latency.mean;
115        if tensor.bandwidth < min_bandwidth { min_bandwidth = tensor.bandwidth; }
116    }
117    let max_entropy = tensors.len() as f64 * 1.0;
118    let stability_score = if max_entropy > 0.0 {
119        (1.0 - entropy / max_entropy).clamp(0.0, 1.0)
120    } else { 1.0 };
121    RouteHealthScore {
122        entropy,
123        stability_score,
124        is_unstable: entropy > 0.5 * tensors.len() as f64,
125        total_latency_ms: total_latency,
126        min_bandwidth_mbps: if min_bandwidth.is_infinite() { 0.0 } else { min_bandwidth },
127    }
128}

130#[derive(Debug, Clone, Serialize, Deserialize)]
131pub struct TriangleCheckResult {
132    pub passed: bool,
133    pub deviation_score: f64,
134    pub violation: Option<String>,
135    pub new_trust_weight: f64,
136}
137
138pub fn triangle_check(l_ab: f64, l_ac: f64, l_bc: f64, current_trust_weight: f64) -> TriangleCheckResult {
139    let lower_bound = (l_ac - l_bc).abs();
140    let upper_bound = l_ac + l_bc;
141    let lower_with_tolerance = lower_bound * (1.0 - TRIANGLE_TOLERANCE);
142    let upper_with_tolerance = upper_bound * (1.0 + TRIANGLE_TOLERANCE);
143    let passed = l_ab >= lower_with_tolerance && l_ab <= upper_with_tolerance;
144    let deviation_score = if passed { 0.0 }
145        else if l_ab < lower_with_tolerance { (lower_with_tolerance - l_ab) / lower_with_tolerance.max(1.0) }
146        else { (l_ab - upper_with_tolerance) / upper_with_tolerance.max(1.0) };
147    let new_trust_weight = if passed {
148        (current_trust_weight * (1.0 + TRUST_DECAY_ALPHA * 0.1)).min(1.0)
149    } else {
150        current_trust_weight * (-TRUST_DECAY_ALPHA * deviation_score).exp()
151    };
152    let violation = if !passed {
153        Some(format!("Triangle violation: L_AB={:.2}ms outside [{:.2}, {:.2}]ms. Deviation: {:.1}%",
154            l_ab, lower_with_tolerance, upper_with_tolerance, deviation_score * 100.0))
155    } else { None };
156    TriangleCheckResult { passed, deviation_score, violation, new_trust_weight }
157}

159#[derive(Debug, Default, Serialize, Deserialize)]
160pub struct TrustRegistry {
161    scores: HashMap<String, f64>,
162    check_counts: HashMap<String, u64>,
163}
164
165impl TrustRegistry {
166    pub fn new() -> Self { Self::default() }
167
168    pub fn get_trust(&self, node_id: &str) -> f64 {
169        *self.scores.get(node_id).unwrap_or(&1.0)
170    }
171
172    pub fn record_check(&mut self, node_id: &str, result: &TriangleCheckResult) {
173        self.scores.insert(node_id.to_string(), result.new_trust_weight);
174        *self.check_counts.entry(node_id.to_string()).or_insert(0) += 1;
175    }
176
177    pub fn penalize_unreachable(&mut self, node_id: &str) {
178        let current = self.get_trust(node_id);
179        let penalized = current * (-TRUST_DECAY_ALPHA * 2.0).exp();
180        self.scores.insert(node_id.to_string(), penalized);
181    }
182
183    pub fn get_quarantined_nodes(&self, threshold: f64) -> Vec<String> {
184        self.scores.iter().filter(|(_, &w)| w < threshold).map(|(id, _)| id.clone()).collect()
185    }
186
187    pub fn stats(&self) -> String {
188        let total = self.scores.len();
189        let high_trust = self.scores.values().filter(|&&w| w > 0.8).count();
190        let quarantined = self.scores.values().filter(|&&w| w < 0.2).count();
191        format!("Nodes: {}. High trust (>0.8): {}. Quarantined (<0.2): {}.", total, high_trust, quarantined)
192    }
193}
