19pub const KEY_SIZE: usize    = 32; // 256 бит
20pub const NONCE_SIZE: usize  = 12; // 96 бит
21pub const TAG_SIZE: usize    = 16; // 128 бит MAC
22pub const BLOCK_SIZE: usize  = 64; // ChaCha20 блок

28fn u32_from_le(b: &[u8], i: usize) -> u32 {
29    u32::from_le_bytes([b[i], b[i+1], b[i+2], b[i+3]])
30}
31
32fn quarter_round(a: &mut u32, b: &mut u32, c: &mut u32, d: &mut u32) {
33    *a = a.wrapping_add(*b); *d ^= *a; *d = d.rotate_left(16);
34    *c = c.wrapping_add(*d); *b ^= *c; *b = b.rotate_left(12);
35    *a = a.wrapping_add(*b); *d ^= *a; *d = d.rotate_left(8);
36    *c = c.wrapping_add(*d); *b ^= *c; *b = b.rotate_left(7);
37}

39// -----------------------------------------------------------------------------
40// ChaCha20 — потоковый шифр RFC 8439
41// -----------------------------------------------------------------------------
42
43pub struct ChaCha20 {
44    state: [u32; 16],
45}
46
47impl ChaCha20 {
48    pub fn new(key: &[u8; KEY_SIZE], nonce: &[u8; NONCE_SIZE], counter: u32) -> Self {
49        // Константа "expand 32-byte k"
50        let mut state = [0u32; 16];
51        state[0]  = 0x61707865;
52        state[1]  = 0x3320646e;
53        state[2]  = 0x79622d32;
54        state[3]  = 0x6b206574;
55        // Ключ (256 бит = 8 u32)
56        for i in 0..8 {
57            state[4+i] = u32_from_le(key, i*4);
58        }
59        // Счётчик
60        state[12] = counter;
61        // Nonce (96 бит = 3 u32)
62        state[13] = u32_from_le(nonce, 0);
63        state[14] = u32_from_le(nonce, 4);
64        state[15] = u32_from_le(nonce, 8);
65        ChaCha20 { state }
66    }
67
68    fn block(&self) -> [u8; BLOCK_SIZE] {
69        let mut x = self.state;
70        // 20 раундов (10 double rounds)
71        for _ in 0..10 {
72            // Колонки — извлекаем во временные переменные
73            let (mut a,mut b,mut c,mut d) = (x[0],x[4],x[8],x[12]);
74            quarter_round(&mut a,&mut b,&mut c,&mut d);
75            x[0]=a;x[4]=b;x[8]=c;x[12]=d;
76
77            let (mut a,mut b,mut c,mut d) = (x[1],x[5],x[9],x[13]);
78            quarter_round(&mut a,&mut b,&mut c,&mut d);
79            x[1]=a;x[5]=b;x[9]=c;x[13]=d;
80
81            let (mut a,mut b,mut c,mut d) = (x[2],x[6],x[10],x[14]);
82            quarter_round(&mut a,&mut b,&mut c,&mut d);
83            x[2]=a;x[6]=b;x[10]=c;x[14]=d;
84
85            let (mut a,mut b,mut c,mut d) = (x[3],x[7],x[11],x[15]);
86            quarter_round(&mut a,&mut b,&mut c,&mut d);
87            x[3]=a;x[7]=b;x[11]=c;x[15]=d;
88
89            // Диагонали
90            let (mut a,mut b,mut c,mut d) = (x[0],x[5],x[10],x[15]);
91            quarter_round(&mut a,&mut b,&mut c,&mut d);
92            x[0]=a;x[5]=b;x[10]=c;x[15]=d;
93
94            let (mut a,mut b,mut c,mut d) = (x[1],x[6],x[11],x[12]);
95            quarter_round(&mut a,&mut b,&mut c,&mut d);
96            x[1]=a;x[6]=b;x[11]=c;x[12]=d;
97
98            let (mut a,mut b,mut c,mut d) = (x[2],x[7],x[8],x[13]);
99            quarter_round(&mut a,&mut b,&mut c,&mut d);
100            x[2]=a;x[7]=b;x[8]=c;x[13]=d;
101
102            let (mut a,mut b,mut c,mut d) = (x[3],x[4],x[9],x[14]);
103            quarter_round(&mut a,&mut b,&mut c,&mut d);
104            x[3]=a;x[4]=b;x[9]=c;x[14]=d;
105        }
106        // Добавляем исходное состояние
107        let mut out = [0u8; BLOCK_SIZE];
108        for i in 0..16 {
109            let word = x[i].wrapping_add(self.state[i]);
110            let bytes = word.to_le_bytes();
111            out[i*4..i*4+4].copy_from_slice(&bytes);
112        }
113        out
114    }
115
116    pub fn keystream(&mut self, len: usize) -> Vec<u8> {
117        let mut result = Vec::with_capacity(len);
118        let mut remaining = len;
119        while remaining > 0 {
120            let block = self.block();
121            let take = remaining.min(BLOCK_SIZE);
122            result.extend_from_slice(&block[..take]);
123            remaining -= take;
124            self.state[12] = self.state[12].wrapping_add(1);
125        }
126        result
127    }
128
129    pub fn encrypt(&mut self, plaintext: &[u8]) -> Vec<u8> {
130        let ks = self.keystream(plaintext.len());
131        plaintext.iter().zip(ks.iter()).map(|(p,k)| p^k).collect()
132    }
133
134    pub fn decrypt(&mut self, ciphertext: &[u8]) -> Vec<u8> {
135        self.encrypt(ciphertext) // XOR симметричен
136    }
137}

139// -----------------------------------------------------------------------------
140// Poly1305 — MAC аутентификатор RFC 8439
141// -----------------------------------------------------------------------------
142
143pub struct Poly1305 {
144    r: [u64; 5],
145    s: [u64; 4],
146    h: [u64; 5],
147}
148
149impl Poly1305 {
150    pub fn new(key: &[u8; 32]) -> Self {
151        // r: первые 16 байт ключа (clamp)
152        let mut r = [0u64; 5];
153        let r128 = u128::from_le_bytes(key[..16].try_into().unwrap());
154        // clamp: обнуляем определённые биты
155        let r_clamped = r128 & 0x0ffffffc0ffffffc0ffffffc0fffffff;
156        r[0] = (r_clamped & 0x3ffffff) as u64;
157        r[1] = ((r_clamped >> 26) & 0x3ffffff) as u64;
158        r[2] = ((r_clamped >> 52) & 0x3ffffff) as u64;
159        r[3] = ((r_clamped >> 78) & 0x3ffffff) as u64;
160        r[4] = ((r_clamped >> 104) & 0x3ffffff) as u64;
161
162        // s: последние 16 байт ключа
163        let mut s = [0u64; 4];
164        for i in 0..4 {
165            s[i] = u32::from_le_bytes(key[16+i*4..20+i*4].try_into().unwrap()) as u64;
166        }
167
168        Poly1305 { r, s, h: [0u64; 5] }
169    }
170
171    fn process_block(&mut self, block: &[u8], last: bool) {
172        let shift = block.len() * 8;
173        let pad = if last || shift >= 128 { 0u128 } else { 1u128 << shift };
174        let mut n = [0u8; 17];
175        n[..block.len()].copy_from_slice(block);
176        if !last { n[block.len()] = 1; }
177        let n = u128::from_le_bytes(n[..16].try_into().unwrap()) | pad;
178
179        // Добавляем к аккумулятору
180        let n0 = (n & 0x3ffffff) as u64;
181        let n1 = ((n >> 26) & 0x3ffffff) as u64;
182        let n2 = ((n >> 52) & 0x3ffffff) as u64;
183        let n3 = ((n >> 78) & 0x3ffffff) as u64;
184        let n4 = ((n >> 104) & 0x3ffffff) as u64;
185
186        let mut h = [0u64; 5];
187        h[0] = self.h[0].wrapping_add(n0);
188        h[1] = self.h[1].wrapping_add(n1);
189        h[2] = self.h[2].wrapping_add(n2);
190        h[3] = self.h[3].wrapping_add(n3);
191        h[4] = self.h[4].wrapping_add(n4);
192
193        // Умножение h * r (упрощённое для pure Rust)
194        let r = &self.r;
195        let mut t = [0u128; 5];
196        t[0] = h[0] as u128 * r[0] as u128
197             + h[1] as u128 * (r[4]*5) as u128
198             + h[2] as u128 * (r[3]*5) as u128
199             + h[3] as u128 * (r[2]*5) as u128
200             + h[4] as u128 * (r[1]*5) as u128;
201        t[1] = h[0] as u128 * r[1] as u128
202             + h[1] as u128 * r[0] as u128
203             + h[2] as u128 * (r[4]*5) as u128
204             + h[3] as u128 * (r[3]*5) as u128
205             + h[4] as u128 * (r[2]*5) as u128;
206        t[2] = h[0] as u128 * r[2] as u128
207             + h[1] as u128 * r[1] as u128
208             + h[2] as u128 * r[0] as u128
209             + h[3] as u128 * (r[4]*5) as u128
210             + h[4] as u128 * (r[3]*5) as u128;
211        t[3] = h[0] as u128 * r[3] as u128
212             + h[1] as u128 * r[2] as u128
213             + h[2] as u128 * r[1] as u128
214             + h[3] as u128 * r[0] as u128
215             + h[4] as u128 * (r[4]*5) as u128;
216        t[4] = h[0] as u128 * r[4] as u128
217             + h[1] as u128 * r[3] as u128
218             + h[2] as u128 * r[2] as u128
219             + h[3] as u128 * r[1] as u128
220             + h[4] as u128 * r[0] as u128;
221
222        // Редукция по 2^130-5
223        let mut h2 = [0u64; 5];
224        let mut c = 0u128;
225        for i in 0..5 {
226            let v = t[i] + c;
227            h2[i] = (v & 0x3ffffff) as u64;
228            c = v >> 26;
229        }
230        h2[0] = h2[0].wrapping_add((c * 5) as u64);
231        c = (h2[0] >> 26) as u128;
232        h2[0] &= 0x3ffffff;
233        h2[1] = h2[1].wrapping_add(c as u64);
234
235        self.h = h2;
236    }
237
238    pub fn mac(&mut self, msg: &[u8]) -> [u8; TAG_SIZE] {
239        // Обрабатываем блоки по 16 байт
240        let mut i = 0;
241        while i + 16 <= msg.len() {
242            let block = &msg[i..i+16];
243            self.process_block(block, false);
244            i += 16;
245        }
246        if i < msg.len() {
247            self.process_block(&msg[i..], true);
248        }
249
250        // Финализация: h + s
251        let mut h = self.h;
252        // Полная редукция
253        let mut c = h[1] >> 26; h[1] &= 0x3ffffff; h[2] = h[2].wrapping_add(c);
254        c = h[2] >> 26; h[2] &= 0x3ffffff; h[3] = h[3].wrapping_add(c);
255        c = h[3] >> 26; h[3] &= 0x3ffffff; h[4] = h[4].wrapping_add(c);
256        c = h[4] >> 26; h[4] &= 0x3ffffff; h[0] = h[0].wrapping_add(c * 5);
257        c = h[0] >> 26; h[0] &= 0x3ffffff; h[1] = h[1].wrapping_add(c);
258
259        // Сборка в 128 бит
260        let tag_val = (h[0] as u128)
261            | ((h[1] as u128) << 26)
262            | ((h[2] as u128) << 52)
263            | ((h[3] as u128) << 78)
264            | ((h[4] as u128) << 104);
265
266        let s = (self.s[0] as u128)
267            | ((self.s[1] as u128) << 32)
268            | ((self.s[2] as u128) << 64)
269            | ((self.s[3] as u128) << 96);
270
271        let tag = tag_val.wrapping_add(s);
272        tag.to_le_bytes()
273    }
274}

284#[derive(Debug, Clone)]
285pub struct AeadCiphertext {
286    pub nonce: [u8; NONCE_SIZE],
287    pub ciphertext: Vec<u8>,
288    pub tag: [u8; TAG_SIZE],
289    pub aad_len: usize,
290}
291
292impl AeadCiphertext {
293    pub fn to_bytes(&self) -> Vec<u8> {
294        let mut out = Vec::with_capacity(NONCE_SIZE + self.ciphertext.len() + TAG_SIZE);
295        out.extend_from_slice(&self.nonce);
296        out.extend_from_slice(&self.ciphertext);
297        out.extend_from_slice(&self.tag);
298        out
299    }
300    pub fn len(&self) -> usize {
301        NONCE_SIZE + self.ciphertext.len() + TAG_SIZE
302    }
303}
304
305impl ChaCha20Poly1305 {
306    pub fn new(key: [u8; KEY_SIZE]) -> Self {
307        ChaCha20Poly1305 { key }
308    }
309
310    // Генерация one-time ключа Poly1305
311    fn poly1305_key(&self, nonce: &[u8; NONCE_SIZE]) -> [u8; 32] {
312        let mut cipher = ChaCha20::new(&self.key, nonce, 0);
313        let ks = cipher.keystream(64);
314        ks[..32].try_into().unwrap()
315    }
316
317    // Паддинг до кратного 16
318    fn pad16(len: usize) -> Vec<u8> {
319        let rem = len % 16;
320        if rem == 0 { vec![] } else { vec![0u8; 16 - rem] }
321    }
322
323    pub fn seal(&self, plaintext: &[u8], aad: &[u8],
324                nonce: &[u8; NONCE_SIZE]) -> AeadCiphertext {
325        // Шифруем с counter=1
326        let mut cipher = ChaCha20::new(&self.key, nonce, 1);
327        let ciphertext = cipher.encrypt(plaintext);
328
329        // MAC: poly1305(aad || pad || ct || pad || len_aad || len_ct)
330        let poly_key = self.poly1305_key(nonce);
331        let mut mac_input = Vec::new();
332        mac_input.extend_from_slice(aad);
333        mac_input.extend_from_slice(&Self::pad16(aad.len()));
334        mac_input.extend_from_slice(&ciphertext);
335        mac_input.extend_from_slice(&Self::pad16(ciphertext.len()));
336        mac_input.extend_from_slice(&(aad.len() as u64).to_le_bytes());
337        mac_input.extend_from_slice(&(ciphertext.len() as u64).to_le_bytes());
338
339        let tag = Poly1305::new(&poly_key).mac(&mac_input);
340
341        AeadCiphertext { nonce: *nonce, ciphertext, tag, aad_len: aad.len() }
342    }
343
344    pub fn open(&self, ct: &AeadCiphertext, aad: &[u8]) -> Result<Vec<u8>, &'static str> {
345        // Верифицируем MAC
346        let poly_key = self.poly1305_key(&ct.nonce);
347        let mut mac_input = Vec::new();
348        mac_input.extend_from_slice(aad);
349        mac_input.extend_from_slice(&Self::pad16(aad.len()));
350        mac_input.extend_from_slice(&ct.ciphertext);
351        mac_input.extend_from_slice(&Self::pad16(ct.ciphertext.len()));
352        mac_input.extend_from_slice(&(aad.len() as u64).to_le_bytes());
353        mac_input.extend_from_slice(&(ct.ciphertext.len() as u64).to_le_bytes());
354
355        let expected_tag = Poly1305::new(&poly_key).mac(&mac_input);
356
357        // Constant-time сравнение
358        let tag_ok = ct.tag.iter().zip(expected_tag.iter())
359            .fold(0u8, |acc, (a,b)| acc | (a^b)) == 0;
360        if !tag_ok { return Err("authentication failed"); }
361
362        // Расшифровываем
363        let mut cipher = ChaCha20::new(&self.key, &ct.nonce, 1);
364        Ok(cipher.decrypt(&ct.ciphertext))
365    }
366}

368// -----------------------------------------------------------------------------
369// X25519 — ECDH обмен ключами (упрощённая Curve25519)
370// -----------------------------------------------------------------------------
371
372pub struct X25519 {
373    rng: u64,
374}
375
376impl X25519 {
377    pub fn new(seed: u64) -> Self { X25519 { rng: seed } }
378
379    fn rand_bytes(&mut self, n: usize) -> Vec<u8> {
380        let mut out = Vec::with_capacity(n);
381        while out.len() < n {
382            self.rng ^= self.rng << 13;
383            self.rng ^= self.rng >> 7;
384            self.rng ^= self.rng << 17;
385            out.extend_from_slice(&self.rng.to_le_bytes());
386        }
387        out.truncate(n);
388        out
389    }
390
391    pub fn generate_keypair(&mut self) -> (Vec<u8>, Vec<u8>) {
392        let mut privkey = self.rand_bytes(KEY_SIZE);
393        // Clamp приватного ключа (Curve25519 требование)
394        privkey[0]  &= 248;
395        privkey[31] &= 127;
396        privkey[31] |= 64;
397        // Публичный ключ = scalar_mult(privkey, basepoint)
398        // Упрощённая версия: используем BLAKE2b-подобный микс
399        let pubkey = self.derive_public(&privkey);
400        (privkey, pubkey)
401    }
402
403    fn derive_public(&self, privkey: &[u8]) -> Vec<u8> {
404        // Детерминированное получение публичного ключа из приватного
405        // (в реальном X25519 — умножение на базовую точку Curve25519)
406        let mut state = 0x43757276_65323535u64; // "Curve255"
407        let mut pub_key = Vec::with_capacity(KEY_SIZE);
408        for (i, &b) in privkey.iter().enumerate() {
409            state ^= (b as u64).wrapping_mul(0x517cc1b7_27220a95);
410            state = state.rotate_left((i % 64) as u32 + 1);
411            state ^= state >> 33;
412            state = state.wrapping_mul(0xff51afd7ed558ccd);
413            state ^= state >> 33;
414            pub_key.push((state & 0xff) as u8);
415        }
416        pub_key
417    }
418
419    pub fn diffie_hellman(privkey: &[u8], pubkey: &[u8]) -> Vec<u8> {
420        // ECDH: shared_secret = scalar_mult(privkey, pubkey)
421        // Упрощённая версия: XOR + mix для демонстрации протокола
422        let mut shared = Vec::with_capacity(KEY_SIZE);
423        let mut state = 0xD415_A4E5_5ECE_7000u64;
424        for i in 0..KEY_SIZE {
425            let a = privkey[i % privkey.len()] as u64;
426            let b = pubkey[i % pubkey.len()] as u64;
427            state = state.wrapping_add(a.wrapping_mul(b));
428            state ^= state >> 17;
429            state = state.wrapping_mul(0xbf58476d1ce4e5b9);
430            state ^= state >> 31;
431            shared.push((state & 0xff) as u8);
432        }
433        shared
434    }
435}

437// -----------------------------------------------------------------------------
438// FederationCipher — высокоуровневый API
439// -----------------------------------------------------------------------------
440
441pub struct FederationCipher {
442    sessions: HashMap<String, ChaCha20Poly1305>,
443    rng: u64,
444    pub encrypt_count: u64,
445    pub decrypt_count: u64,
446    pub bytes_encrypted: u64,
447    pub auth_failures: u64,
448}
449
450impl FederationCipher {
451    pub fn new() -> Self {
452        FederationCipher {
453            sessions: HashMap::new(),
454            rng: 0xFEDC_1A50_C0DE_0000,
455            encrypt_count: 0, decrypt_count: 0,
456            bytes_encrypted: 0, auth_failures: 0,
457        }
458    }
459
460    fn next_rng(&mut self) -> u64 {
461        self.rng ^= self.rng << 13;
462        self.rng ^= self.rng >> 7;
463        self.rng ^= self.rng << 17;
464        self.rng
465    }
466
467    pub fn random_nonce(&mut self) -> [u8; NONCE_SIZE] {
468        let mut n = [0u8; NONCE_SIZE];
469        for i in 0..3 {
470            let r = self.next_rng().to_le_bytes();
471            n[i*4..i*4+4].copy_from_slice(&r[..4]);
472        }
473        n
474    }
475
476    pub fn establish_session(&mut self, peer_id: &str, shared_secret: &[u8]) {
477        let mut key = [0u8; KEY_SIZE];
478        for i in 0..KEY_SIZE {
479            key[i] = shared_secret[i % shared_secret.len()];
480        }
481        // HKDF-подобный mix
482        for i in 0..KEY_SIZE {
483            let r = self.next_rng();
484            key[i] ^= (r >> (i*3 % 57)) as u8;
485        }
486        // Восстанавливаем детерминированно из shared_secret
487        let mut key2 = [0u8; KEY_SIZE];
488        let mut state = 0x4845_4453_4554u64; // "HEDSET"
489        for i in 0..KEY_SIZE {
490            state ^= shared_secret[i % shared_secret.len()] as u64;
491            state = state.wrapping_mul(6364136223846793005);
492            state = state.wrapping_add(1442695040888963407);
493            key2[i] = (state >> 33) as u8;
494        }
495        self.sessions.insert(peer_id.to_string(), ChaCha20Poly1305::new(key2));
496    }
497
498    pub fn encrypt_pulse(&mut self, peer_id: &str, pulse: &[u8],
499                          aad: &[u8]) -> Option<AeadCiphertext> {
500        let nonce = self.random_nonce();
501        let cipher = self.sessions.get(peer_id)?;
502        let ct = cipher.seal(pulse, aad, &nonce);
503        self.encrypt_count += 1;
504        self.bytes_encrypted += pulse.len() as u64;
505        Some(ct)
506    }
507
508    pub fn decrypt_pulse(&mut self, peer_id: &str, ct: &AeadCiphertext,
509                          aad: &[u8]) -> Result<Vec<u8>, &'static str> {
510        let cipher = self.sessions.get(peer_id)
511            .ok_or("no session")?;
512        match cipher.open(ct, aad) {
513            Ok(pt) => { self.decrypt_count += 1; Ok(pt) }
514            Err(e) => { self.auth_failures += 1; Err(e) }
515        }
516    }
517
518    pub fn stats(&self) -> CipherStats {
519        CipherStats {
520            sessions: self.sessions.len(),
521            encrypt_count: self.encrypt_count,
522            decrypt_count: self.decrypt_count,
523            bytes_encrypted: self.bytes_encrypted,
524            auth_failures: self.auth_failures,
525        }
526    }
527}
