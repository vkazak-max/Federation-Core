20pub const MIN_BID_CREDITS: f64    = 0.1;
21pub const MARKET_FEE_RATE: f64    = 0.02;  // 2% комиссия рынка
22pub const SLASHING_RATE: f64      = 0.20;  // 20% штраф за провал
23pub const PREMIUM_THRESHOLD: f64  = 2.0;   // выше — «премиум» трафик

26// -----------------------------------------------------------------------------
27// TrafficTier — класс трафика
28// -----------------------------------------------------------------------------
29
30#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
31pub enum TrafficTier {
32    Economy,    // дёшево, без гарантий
33    Standard,   // средняя цена, базовые гарантии
34    Premium,    // дорого, гарантированная доставка
35    Armored,    // максимум — StandoffDecoy обязателен
36}
37
38impl TrafficTier {
39    pub fn base_price(&self) -> f64 {
40        match self {
41            TrafficTier::Economy  => 0.1,
42            TrafficTier::Standard => 0.5,
43            TrafficTier::Premium  => 2.0,
44            TrafficTier::Armored  => 5.0,
45        }
46    }
47    pub fn name(&self) -> &str {
48        match self {
49            TrafficTier::Economy  => "Economy",
50            TrafficTier::Standard => "Standard",
51            TrafficTier::Premium  => "Premium",
52            TrafficTier::Armored  => "Armored",
53        }
54    }
55    pub fn requires_tactic(&self) -> Option<&str> {
56        match self {
57            TrafficTier::Armored  => Some("StandoffDecoy"),
58            TrafficTier::Premium  => Some("CumulativeStrike"),
59            _                     => None,
60        }
61    }
62}

68#[derive(Debug, Clone, Serialize, Deserialize)]
69pub struct BidRequest {
70    pub bid_id: u64,
71    pub user_id: String,
72    pub destination_region: String,
73    pub payload_size_kb: u32,
74    pub max_price: f64,          // максимум credits готов платить
75    pub tier: TrafficTier,
76    pub deadline_ms: u64,
77    pub submitted_at: i64,
78}
79
80impl BidRequest {
81    pub fn new(user_id: &str, region: &str, size_kb: u32,
82               max_price: f64, tier: TrafficTier, counter: u64) -> Self {
83        use std::time::{SystemTime, UNIX_EPOCH};
84        let now = SystemTime::now().duration_since(UNIX_EPOCH)
85            .unwrap().as_millis() as i64;
86        BidRequest {
87            bid_id: counter,
88            user_id: user_id.to_string(),
89            destination_region: region.to_string(),
90            payload_size_kb: size_kb,
91            max_price,
92            tier,
93            deadline_ms: AUCTION_WINDOW_MS,
94            submitted_at: now,
95        }
96    }
97}

103#[derive(Debug, Clone, Serialize, Deserialize)]
104pub struct NodeOffer {
105    pub offer_id: u64,
106    pub node_id: String,
107    pub bid_id: u64,
108    pub price: f64,              // credits за доставку
109    pub tactic: String,
110    pub estimated_latency_ms: u32,
111    pub success_guarantee: f64,  // вероятность успеха (0..1)
112    pub stake: f64,              // залог (будет slash при провале)
113    pub region_difficulty: f64,
114}
115
116impl NodeOffer {
117    pub fn score(&self) -> f64 {
118        // Скоринг: дешевле + надёжнее + быстрее = лучше
119        let price_score    = 1.0 / self.price.max(0.001);
120        let quality_score  = self.success_guarantee;
121        let latency_score  = 1.0 / (self.estimated_latency_ms as f64).max(1.0);
122        let tactic_bonus   = match self.tactic.as_str() {
123            "AikiReflection"   => 1.3,
124            "CumulativeStrike" => 1.2,
125            "StandoffDecoy"    => 1.1,
126            "Hybrid"           => 1.25,
127            _                  => 1.0,
128        };
129        (price_score * 0.4 + quality_score * 0.4 + latency_score * 0.2)
130            * tactic_bonus
131    }
132}

166// -----------------------------------------------------------------------------
167// BandwidthMarket — главный объект рынка
168// -----------------------------------------------------------------------------
169
170pub struct BandwidthMarket {
171    pub bids: HashMap<u64, BidRequest>,
172    pub offers: HashMap<u64, Vec<NodeOffer>>,  // bid_id → offers
173    pub results: Vec<AuctionResult>,
174    pub settlements: Vec<Settlement>,
175    pub node_balances: HashMap<String, f64>,
176    pub market_treasury: f64,
177    pub total_volume: f64,
178    counter: u64,
179}
180
181impl BandwidthMarket {
182    pub fn new() -> Self {
183        BandwidthMarket {
184            bids: HashMap::new(),
185            offers: HashMap::new(),
186            results: vec![],
187            settlements: vec![],
188            node_balances: HashMap::new(),
189            market_treasury: 0.0,
190            total_volume: 0.0,
191            counter: 0,
192        }
193    }
194
195    pub fn submit_bid(&mut self, user_id: &str, region: &str,
196                      size_kb: u32, max_price: f64,
197                      tier: TrafficTier) -> u64 {
198        self.counter += 1;
199        let bid = BidRequest::new(user_id, region, size_kb,
200            max_price, tier, self.counter);
201        let id = bid.bid_id;
202        self.bids.insert(id, bid);
203        id
204    }
205
206    pub fn submit_offer(&mut self, node_id: &str, bid_id: u64,
207                        price: f64, tactic: &str,
208                        latency_ms: u32, guarantee: f64,
209                        stake: f64, difficulty: f64) {
210        self.counter += 1;
211        let offer = NodeOffer {
212            offer_id: self.counter,
213            node_id: node_id.to_string(),
214            bid_id, price, tactic: tactic.to_string(),
215            estimated_latency_ms: latency_ms,
216            success_guarantee: guarantee,
217            stake, region_difficulty: difficulty,
218        };
219        self.offers.entry(bid_id).or_default().push(offer);
220    }
221
222    /// Провести аукцион — выбрать победителя
223    pub fn run_auction(&mut self, bid_id: u64) -> Option<AuctionResult> {
224        let bid = self.bids.get(&bid_id)?.clone();
225        let offers = self.offers.get(&bid_id)?;
226
227        // Фильтруем: цена <= max_price
228        let mut valid: Vec<&NodeOffer> = offers.iter()
229            .filter(|o| o.price <= bid.max_price)
230            .collect();
231
232        // Проверяем требования тактики для тира
233        if let Some(required_tactic) = bid.tier.requires_tactic() {
234            valid.retain(|o| o.tactic == required_tactic
235                || o.tactic == "Hybrid");
236        }
237
238        if valid.is_empty() { return None; }
239
240        // Победитель — максимальный score
241        let winner = valid.iter()
242            .max_by(|a, b| a.score().partial_cmp(&b.score()).unwrap())?;
243
244        let market_fee = winner.price * MARKET_FEE_RATE;
245        let node_revenue = winner.price - market_fee;
246
247        let result = AuctionResult {
248            bid_id,
249            winner_node: winner.node_id.clone(),
250            winning_price: winner.price,
251            winning_tactic: winner.tactic.clone(),
252            competing_offers: offers.len(),
253            market_fee,
254            node_revenue,
255            success_guarantee: winner.success_guarantee,
256        };
257
258        self.results.push(result.clone());
259        self.total_volume += winner.price;
260        self.market_treasury += market_fee;
261        Some(result)
262    }
263
264    /// Расчёт после доставки
265    pub fn settle(&mut self, bid_id: u64, node_id: &str,
266                  delivered: bool, agreed_price: f64,
267                  stake: f64) -> Settlement {
268        let (actual_paid, slash, reason) = if delivered {
269            (agreed_price, 0.0, "Доставлено успешно".into())
270        } else {
271            let slash = stake * SLASHING_RATE;
272            (0.0, slash, "Провал доставки — slash".into())
273        };
274
275        let net = actual_paid - slash;
276        *self.node_balances.entry(node_id.to_string()).or_insert(0.0) += net;
277
278        let s = Settlement {
279            bid_id, node_id: node_id.to_string(),
280            delivered, agreed_price, actual_paid,
281            slash_amount: slash, net_earnings: net, reason,
282        };
283        self.settlements.push(s.clone());
284        s
285    }
286
287    pub fn price_discovery(&self, region: &str) -> PriceStats {
288        let relevant: Vec<&AuctionResult> = self.results.iter()
289            .filter(|r| {
290                self.bids.get(&r.bid_id)
291                    .map(|b| b.destination_region == region)
292                    .unwrap_or(false)
293            }).collect();
294
295        if relevant.is_empty() {
296            return PriceStats { region: region.to_string(),
297                min: 0.0, max: 0.0, avg: 0.0, count: 0 };
298        }
299
300        let prices: Vec<f64> = relevant.iter()
301            .map(|r| r.winning_price).collect();
302        PriceStats {
303            region: region.to_string(),
304            min: prices.iter().cloned().fold(f64::MAX, f64::min),
305            max: prices.iter().cloned().fold(f64::MIN, f64::max),
306            avg: prices.iter().sum::<f64>() / prices.len() as f64,
307            count: prices.len(),
308        }
309    }
310
311    pub fn market_stats(&self) -> MarketStats {
312        let total_bids = self.bids.len();
313        let matched = self.results.len();
314        let fill_rate = if total_bids > 0 {
315            matched as f64 / total_bids as f64 } else { 0.0 };
316
317        MarketStats {
318            total_bids, matched_bids: matched,
319            fill_rate, total_volume: self.total_volume,
320            market_treasury: self.market_treasury,
321            successful_deliveries: self.settlements.iter()
322                .filter(|s| s.delivered).count(),
323            total_slashed: self.settlements.iter()
324                .map(|s| s.slash_amount).sum(),
325        }
326    }
327}
