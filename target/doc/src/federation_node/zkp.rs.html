28/// Размер ключа шифрования (байт)
29pub const KEY_SIZE: usize = 32;
30
31/// Размер nonce (байт)
32pub const NONCE_SIZE: usize = 16;

34/// Максимальное число слоёв onion
35pub const MAX_ONION_LAYERS: usize = 8;

37// -----------------------------------------------------------------------------
38// Простой PRNG на основе xorshift64 (без внешних зависимостей)
39// -----------------------------------------------------------------------------
40
41fn xorshift64(state: &mut u64) -> u64 {
42    *state ^= *state << 13;
43    *state ^= *state >> 7;
44    *state ^= *state << 17;
45    *state
46}
47
48/// Генерация псевдослучайных байт из seed
49fn generate_bytes(seed: u64, len: usize) -> Vec<u8> {
50    let mut state = seed;
51    let mut result = Vec::with_capacity(len);
52    while result.len() < len {
53        let val = xorshift64(&mut state);
54        for b in val.to_le_bytes() {
55            result.push(b);
56            if result.len() == len { break; }
57        }
58    }
59    result
60}

62/// Простой hash (FNV-1a 64bit) — детерминированный
63pub fn fnv_hash(data: &[u8]) -> u64 {
64    let mut hash: u64 = 0xcbf29ce484222325;
65    for &b in data {
66        hash ^= b as u64;
67        hash = hash.wrapping_mul(0x100000001b3);
68    }
69    hash
70}
71
72/// Hex-строка из байт
73fn to_hex(bytes: &[u8]) -> String {
74    bytes.iter().map(|b| format!("{:02x}", b)).collect()
75}

81/// Сессионный ключ для шифрования одного слоя onion
82#[derive(Debug, Clone, Serialize, Deserialize)]
83pub struct SessionKey {
84    pub key_bytes: Vec<u8>,
85    pub nonce: Vec<u8>,
86    pub key_id: String,
87}
88
89impl SessionKey {
90    /// Генерировать новый случайный ключ
91    pub fn generate() -> Self {
92        use std::time::{SystemTime, UNIX_EPOCH};
93        let seed = SystemTime::now()
94            .duration_since(UNIX_EPOCH)
95            .unwrap()
96            .as_nanos() as u64;
97
98        let key_bytes = generate_bytes(seed, KEY_SIZE);
99        let nonce = generate_bytes(seed ^ 0xdeadbeef, NONCE_SIZE);
100        let key_id = to_hex(&generate_bytes(seed ^ 0xcafe, 8));
101
102        SessionKey { key_bytes, nonce, key_id }
103    }
104
105    /// Создать ключ из общего секрета (Diffie-Hellman симуляция)
106    /// В production: X25519 ECDH
107    pub fn from_shared_secret(our_secret: u64, their_public: u64) -> Self {
108        let shared = our_secret.wrapping_mul(their_public) ^ 0x9e3779b97f4a7c15;
109        let key_bytes = generate_bytes(shared, KEY_SIZE);
110        let nonce = generate_bytes(shared ^ 0xdeadbeef, NONCE_SIZE);
111        let key_id = to_hex(&generate_bytes(shared ^ 0xcafe, 8));
112        SessionKey { key_bytes, nonce, key_id }
113    }
114
115    /// XOR-шифрование (симуляция ChaCha20)
116    /// В production заменить на chacha20-poly1305 crate
117    pub fn encrypt(&self, plaintext: &[u8]) -> Vec<u8> {
118        let keystream = generate_bytes(
119            fnv_hash(&self.key_bytes) ^ fnv_hash(&self.nonce),
120            plaintext.len(),
121        );
122        plaintext.iter().zip(keystream.iter()).map(|(p, k)| p ^ k).collect()
123    }
124
125    pub fn decrypt(&self, ciphertext: &[u8]) -> Vec<u8> {
126        // XOR симметричен: decrypt = encrypt
127        self.encrypt(ciphertext)
128    }
129}

131// -----------------------------------------------------------------------------
132// ZkpCommitment — криптографическое обязательство
133// -----------------------------------------------------------------------------
134
135/// Commitment scheme: commit(value, blinding_factor) = hash(value || blinding)
136///
137/// Свойства:
138///   - Hiding:  по commitment нельзя узнать value
139///   - Binding: нельзя открыть commitment с другим value
140#[derive(Debug, Clone, Serialize, Deserialize)]
141pub struct ZkpCommitment {
142    /// Публичный commitment (можно показывать всем)
143    pub commitment: String,
144    /// Blinding factor (держим в секрете до reveal)
145    pub blinding_factor: u64,
146    /// Хеш от value (для верификации при reveal)
147    value_hash: u64,
148}
149
150impl ZkpCommitment {
151    /// Создать commitment для значения
152    pub fn commit(value: &[u8]) -> Self {
153        use std::time::{SystemTime, UNIX_EPOCH};
154        let blinding_factor = SystemTime::now()
155            .duration_since(UNIX_EPOCH)
156            .unwrap()
157            .as_nanos() as u64 ^ 0xfeedface;
158
159        let value_hash = fnv_hash(value);
160
161        // commitment = hash(value || blinding_factor)
162        let mut combined = value.to_vec();
163        combined.extend_from_slice(&blinding_factor.to_le_bytes());
164        let commitment = to_hex(&generate_bytes(fnv_hash(&combined), 32));
165
166        ZkpCommitment { commitment, blinding_factor, value_hash }
167    }
168
169    /// Верифицировать что value соответствует commitment
170    pub fn verify(&self, value: &[u8]) -> bool {
171        let claimed_hash = fnv_hash(value);
172        if claimed_hash != self.value_hash {
173            return false;
174        }
175        let mut combined = value.to_vec();
176        combined.extend_from_slice(&self.blinding_factor.to_le_bytes());
177        let expected = to_hex(&generate_bytes(fnv_hash(&combined), 32));
178        expected == self.commitment
179    }
180}

182// -----------------------------------------------------------------------------
183// OnionLayer — один слой луковичного шифрования
184// -----------------------------------------------------------------------------
185
186/// Один слой onion-маршрутизации.
187/// Каждый узел видит только:
188///   - Зашифрованный payload для следующего хопа
189///   - ID следующего узла (в открытом виде — только для маршрутизации)
190#[derive(Debug, Clone, Serialize, Deserialize)]
191pub struct OnionLayer {
192    /// ID следующего узла (единственное что видит текущий узел)
193    pub next_hop: String,
194    /// Зашифрованный payload (остаток луковицы)
195    pub encrypted_payload: Vec<u8>,
196    /// Commitment для верификации целостности
197    pub integrity_commitment: String,
198    /// Nullifier — предотвращает replay
199    pub nullifier: String,
200}

206/// Полный onion-пакет. Видимая часть минимальна.
207#[derive(Debug, Clone, Serialize, Deserialize)]
208pub struct OnionPacket {
209    /// Только первый слой виден текущему узлу
210    pub outer_layer: OnionLayer,
211    /// Число слоёв (без раскрытия маршрута)
212    pub layer_count: usize,
213    /// Публичный commitment отправителя (без раскрытия ID)
214    pub sender_commitment: String,
215    /// Timestamp для anti-replay
216    pub created_at: i64,
217}

219// -----------------------------------------------------------------------------
220// OnionBuilder — строит многослойный зашифрованный пакет
221// -----------------------------------------------------------------------------
222
223/// Строитель onion-пакета.
224///
225/// Использование:
226///   1. Задаём маршрут [node1, node2, ..., destination]
227///   2. Для каждого узла генерируем SessionKey
228///   3. Оборачиваем payload в слои (от последнего к первому)
229///   4. Каждый узел может снять только свой слой
230pub struct OnionBuilder {
231    route: Vec<String>,
232    keys: Vec<SessionKey>,
233}
234
235impl OnionBuilder {
236    pub fn new() -> Self {
237        OnionBuilder { route: vec![], keys: vec![] }
238    }
239
240    /// Задать маршрут и автоматически сгенерировать ключи
241    pub fn with_route(mut self, route: Vec<String>) -> Self {
242        self.keys = route.iter().map(|_| SessionKey::generate()).collect();
243        self.route = route;
244        self
245    }
246
247    /// Собрать onion-пакет с payload
248    pub fn build(self, payload: &[u8]) -> Result<(OnionPacket, Vec<SessionKey>), String> {
249        if self.route.is_empty() {
250            return Err("Маршрут не задан".to_string());
251        }
252        if self.route.len() > MAX_ONION_LAYERS {
253            return Err(format!("Слишком много слоёв: {} > {}", self.route.len(), MAX_ONION_LAYERS));
254        }
255
256        let layer_count = self.route.len();
257
258        // Строим слои от последнего к первому (как настоящий onion)
259        let mut current_payload = payload.to_vec();
260
261        // Сохраним все слои для дебага
262        let mut layers: Vec<OnionLayer> = Vec::new();
263
264        for i in (0..self.route.len()).rev() {
265            let key = &self.keys[i];
266            let next_hop = if i + 1 < self.route.len() {
267                self.route[i + 1].clone()
268            } else {
269                "DESTINATION".to_string()
270            };
271
272            // Шифруем текущий payload ключом этого узла
273            let encrypted = key.encrypt(&current_payload);
274
275            // Integrity commitment
276            let integrity = to_hex(&generate_bytes(fnv_hash(&encrypted), 16));
277
278            // Nullifier — уникальный идентификатор для anti-replay
279            let nullifier = to_hex(&generate_bytes(
280                fnv_hash(key.key_id.as_bytes()) ^ i as u64,
281                16,
282            ));
283
284            let layer = OnionLayer {
285                next_hop,
286                encrypted_payload: encrypted.clone(),
287                integrity_commitment: integrity,
288                nullifier,
289            };
290
291            layers.push(layer);
292            // Следующий слой оборачивает текущий зашифрованный payload
293            current_payload = encrypted;
294        }
295
296        layers.reverse();
297
298        // Commitment отправителя (без раскрытия ID)
299        let sender_commitment = to_hex(&generate_bytes(
300            fnv_hash(self.route[0].as_bytes()) ^ 0xdeadcafe,
301            32,
302        ));
303
304        use std::time::{SystemTime, UNIX_EPOCH};
305        let now = SystemTime::now()
306            .duration_since(UNIX_EPOCH)
307            .unwrap()
308            .as_millis() as i64;
309
310        let packet = OnionPacket {
311            outer_layer: layers.into_iter().next().unwrap(),
312            layer_count,
313            sender_commitment,
314            created_at: now,
315        };
316
317        Ok((packet, self.keys))
318    }
319}
320
321impl Default for OnionBuilder {
322    fn default() -> Self { Self::new() }
323}

329/// Результат обработки onion-пакета
330#[derive(Debug)]
331pub struct PeeledLayer {
332    /// Следующий узел куда переслать
333    pub next_hop: String,
334    /// Расшифрованный payload для следующего слоя
335    pub inner_payload: Vec<u8>,
336    /// Это последний слой (мы — получатель)?
337    pub is_final: bool,
338}
339
340/// Снять один слой onion-пакета.
341///
342/// Узел знает только свой ключ → видит только next_hop.
343/// Содержимое payload остаётся зашифрованным для него.
344pub fn peel_onion_layer(layer: &OnionLayer, key: &SessionKey) -> PeeledLayer {
345    let inner = key.decrypt(&layer.encrypted_payload);
346    let is_final = layer.next_hop == "DESTINATION";
347    PeeledLayer {
348        next_hop: layer.next_hop.clone(),
349        inner_payload: inner,
350        is_final,
351    }
352}

354// -----------------------------------------------------------------------------
355// NullifierSet — защита от replay-атак
356// -----------------------------------------------------------------------------
357
358/// Реестр использованных nullifiers.
359/// Если nullifier уже видели — пакет отброшен (replay-атака).
360#[derive(Debug, Default)]
361pub struct NullifierSet {
362    seen: HashSet<String>,
363    /// Счётчик отброшенных replay
364    pub replay_attempts: u64,
365}
366
367impl NullifierSet {
368    pub fn new() -> Self { Self::default() }
369
370    /// Проверить и добавить nullifier. Возвращает false если replay.
371    pub fn check_and_add(&mut self, nullifier: &str) -> bool {
372        if self.seen.contains(nullifier) {
373            self.replay_attempts += 1;
374            false
375        } else {
376            self.seen.insert(nullifier.to_string());
377            true
378        }
379    }
380
381    pub fn size(&self) -> usize { self.seen.len() }
382}

384// -----------------------------------------------------------------------------
385// BlindedSsau — SSAU данные без раскрытия узла-отправителя
386// -----------------------------------------------------------------------------
387
388/// SSAU тензор с ослеплённым (blinded) источником.
389/// Промежуточный узел видит данные о качестве канала,
390/// но не знает кто их отправил.
391#[derive(Debug, Clone, Serialize, Deserialize)]
392pub struct BlindedSsau {
393    /// Commitment вместо реального node_id отправителя
394    pub sender_commitment: String,
395    /// Данные канала (в открытом виде — нужны для маршрутизации)
396    pub latency_ms: f64,
397    pub bandwidth_mbps: f64,
398    /// Reliability зашифрован — только получатель может расшифровать
399    pub encrypted_reliability: Vec<u8>,
400    /// Proof что данные валидны (без раскрытия источника)
401    pub validity_proof: String,
402}
403
404impl BlindedSsau {
405    /// Создать ослеплённый SSAU из реального тензора
406    pub fn from_tensor(
407        sender_id: &str,
408        latency_ms: f64,
409        bandwidth_mbps: f64,
410        reliability: f64,
411        session_key: &SessionKey,
412    ) -> Self {
413        // Commitment вместо реального ID
414        let commitment = ZkpCommitment::commit(sender_id.as_bytes());
415
416        // Шифруем reliability (чувствительный параметр)
417        let reliability_bytes = reliability.to_le_bytes();
418        let encrypted_reliability = session_key.encrypt(&reliability_bytes);
419
420        // Proof валидности (в реальной системе — zk-SNARK)
421        let validity_proof = to_hex(&generate_bytes(
422            fnv_hash(sender_id.as_bytes()) ^ (latency_ms as u64),
423            32,
424        ));
425
426        BlindedSsau {
427            sender_commitment: commitment.commitment,
428            latency_ms,
429            bandwidth_mbps,
430            encrypted_reliability,
431            validity_proof,
432        }
433    }
434
435    /// Расшифровать reliability (только если знаешь ключ)
436    pub fn decrypt_reliability(&self, session_key: &SessionKey) -> f64 {
437        let decrypted = session_key.decrypt(&self.encrypted_reliability);
438        if decrypted.len() >= 8 {
439            let mut bytes = [0u8; 8];
440            bytes.copy_from_slice(&decrypted[..8]);
441            f64::from_le_bytes(bytes)
442        } else {
443            0.0
444        }
445    }
446}
