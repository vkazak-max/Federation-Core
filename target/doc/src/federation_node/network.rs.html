148impl PacketBuilder {
149    pub fn new(sender_id: &str) -> Self {
150        Self { sender_id: sender_id.to_string(), recipient_id: None, ttl: 64 }

153    pub fn to(mut self, recipient_id: &str) -> Self {
154        self.recipient_id = Some(recipient_id.to_string());
155        self
156    }

158    pub fn with_ttl(mut self, ttl: u8) -> Self {
159        self.ttl = ttl;
160        self
161    }
162
163    pub fn build(self, message: FederationMessage) -> FederationPacket {
164        use std::time::{SystemTime, UNIX_EPOCH};
165        let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_millis() as i64;
166        let payload_json = serde_json::to_string(&message).unwrap_or_default();
167        let payload_size = payload_json.len() as u32;
168        FederationPacket {
169            header: PacketHeader {
170                packet_id: Uuid::new_v4().to_string(),
171                sender_id: self.sender_id,
172                recipient_id: self.recipient_id,
173                created_at: now,
174                protocol_version: PROTOCOL_VERSION,
175                ttl: self.ttl,
176                payload_size,
177            },
178            message,
179            signature: "placeholder_sig_v1".to_string(),
180        }
181    }
182}

192pub type PacketSender = mpsc::Sender<FederationPacket>;
193pub type PacketReceiver = mpsc::Receiver<FederationPacket>;

195pub struct NodeContext {
196    pub node_id: String,
197    pub inbox: PacketReceiver,
198    pub outbox: PacketSender,
199    pub peers: HashMap<String, PacketSender>,
200    pub packets_processed: u64,
201    pub packets_sent: u64,
202}
203
204impl NodeContext {
205    pub fn new(node_id: &str) -> (Self, PacketSender) {
206        let (inbox_tx, inbox_rx) = mpsc::channel::<FederationPacket>(256);
207        let (outbox_tx, _outbox_rx) = mpsc::channel::<FederationPacket>(256);
208        let ctx = NodeContext {
209            node_id: node_id.to_string(),
210            inbox: inbox_rx,
211            outbox: outbox_tx,
212            peers: HashMap::new(),
213            packets_processed: 0,
214            packets_sent: 0,
215        };
216        (ctx, inbox_tx)
217    }
218
219    pub fn add_peer(&mut self, peer_id: &str, sender: PacketSender) {
220        self.peers.insert(peer_id.to_string(), sender);
221    }
222
223    pub async fn send_to_peer(&mut self, peer_id: &str, packet: FederationPacket) -> Result<(), String> {
224        match self.peers.get(peer_id) {
225            Some(sender) => {
226                sender.send(packet).await.map_err(|e| format!("Send error: {}", e))?;
227                self.packets_sent += 1;
228                Ok(())
229            }
230            None => Err(format!("Peer {} not found", peer_id)),
231        }
232    }
233
234    pub async fn broadcast(&mut self, packet: FederationPacket) -> usize {
235        let mut sent = 0;
236        let peer_ids: Vec<String> = self.peers.keys().cloned().collect();
237        for peer_id in peer_ids {
238            let mut p = packet.clone();
239            p.header.recipient_id = Some(peer_id.clone());
240            if let Some(sender) = self.peers.get(&peer_id) {
241                if sender.send(p).await.is_ok() {
242                    sent += 1;
243                    self.packets_sent += 1;
244                }
245            }
246        }
247        sent
248    }
249
250    pub async fn process_next(&mut self) -> Option<FederationPacket> {
251        match self.inbox.try_recv() {
252            Ok(packet) => { self.packets_processed += 1; Some(packet) }
253            Err(_) => None,
254        }
255    }
256}

258pub fn create_handshake_packet(our_node_id: &str, their_node_id: &str, public_key: &str, known_peers: u32) -> FederationPacket {
259    use std::time::{SystemTime, UNIX_EPOCH};
260    let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_millis() as i64;
261    PacketBuilder::new(our_node_id).to(their_node_id).build(FederationMessage::Handshake(HandshakePayload {
262        node_id: our_node_id.to_string(),
263        protocol_version: PROTOCOL_VERSION,
264        public_key: public_key.to_string(),
265        capabilities: NodeCapabilities {
266            is_relay: true,
267            max_bandwidth_mbps: 100,
268            supports_storage: false,
269            supports_consensus: false,
270        },
271        timestamp: now,
272        known_peers_count: known_peers,
273    }))
274}

276pub fn create_ssau_update_packet(our_node_id: &str, tensors: &[&SsauTensor], sequence: u64) -> FederationPacket {
277    let tensor_messages: Vec<SsauTensorMessage> = tensors.iter().map(|t| SsauTensorMessage::from(*t)).collect();
278    PacketBuilder::new(our_node_id).build(FederationMessage::SsauUpdate(SsauUpdatePayload {
279        reporter_node_id: our_node_id.to_string(),
280        tensors: tensor_messages,
281        sequence,
